defined_text = {
	name = OverclanAcademyGetUniqueStatus
	random = no
	text = {
		localisation_key = overclan_academy_status_built_tt
		trigger = {
			has_unique_voltoj_building = yes
		}
	}
	text = {
		localisation_key = overclan_academy_status_no_academy_tt
		trigger = {
			NOT = { has_voltoj_academy = yes }
		}
	}
	text = {
		localisation_key = overclan_academy_status_ready_tt
		trigger = {
			NOT = { has_unique_voltoj_building = yes }
			overclan_unique_upgrade_available = yes
			voltoj_can_afford_upgrade_unique = yes
		}
	}
	text = {
		localisation_key = overclan_academy_status_unavailable_tt
		trigger = {
			NOT = { has_unique_voltoj_building = yes }
			NOT = { overclan_unique_upgrade_available = yes }
		}
	}
	text = {
		localisation_key = overclan_academy_status_unaffordable_tt
		trigger = {
			NOT = { has_unique_voltoj_building = yes }
			overclan_unique_upgrade_available = yes
			NOT = { voltoj_can_afford_upgrade_unique = yes }
		}
	}
}
