# Scripted localisation for assembling gene pillar clues.
# GetManagoPokuJourneyDesc: selects loc key based on whatever pillar was just explored
# GetMPJHeader_[Pillar]: Gets the event preamble for when there are unvisited nearby gene pillars
# GetMPJClue_[Target]_from[Source]: Gets the clue for going to Target from Source

defined_text = {
	name = GetManagoPokuJourneyDesc
    random = no
	text = {
		trigger = { has_country_flag = journey_continues_from_forest }
		localisation_key = manago_poku.100.from_forest
	}
	text = {
		trigger = { has_country_flag = journey_continues_from_cape }
		localisation_key = manago_poku.100.from_cape
	}
	text = {
		trigger = { has_country_flag = journey_continues_from_jungle }

	}
	text = {

	}
	text = {
		trigger = { has_country_flag = journey_continues_from_fingers }

	}
	text = {
		trigger = { has_country_flag = journey_continues_from_lowlands }
		localisation_key = manago_poku.100.from_lowlands
	}
	text = {
		trigger = { has_country_flag = journey_continues_from_mangroves }

# Preamble when any neighbour is unvisited or hollowness otherwise.

# Forest (White) neighbours: Cape (Yellow)
defined_text = {
	name = GetMPJHeader_Forest
    random = no
	text = {
		trigger = { NOT = { has_country_flag = visited_pillar_cape } }
		localisation_key = manago_poku.100.preamble
	}
	text = {
		trigger = { always = yes }
		localisation_key = manago_poku.100.hollowness
	}
}

# Cape (Yellow) neighbours: Forest (White), Jungle (Orange)
defined_text = {
	name = GetMPJHeader_Cape
    random = no
	text = {
		trigger = {
			OR = {
				NOT = { has_country_flag = visited_pillar_forest }
				NOT = { has_country_flag = visited_pillar_jungle }
			}
		}
		localisation_key = manago_poku.100.preamble
	}
	text = {
		trigger = { always = yes }
		localisation_key = manago_poku.100.hollowness
	}
}
