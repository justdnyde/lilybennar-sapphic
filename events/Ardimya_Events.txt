#Birth of Ardimyan Culture
country_event = {
	id = ardimya.1
	title = ardimya.1.t
	desc = ardimya.1.d
	picture = COLONIZATION_eventPicture
	
	is_triggered_only = yes
	fire_only_once = yes
	
	major = yes
	major_trigger = {
		capital_scope = { continent = europe }
		num_of_colonists = 1
		any_owned_province = {
			region = ardimya_region
		}
		NOT = { has_global_flag = ardimya_event_chain_started } 
	}
	
	trigger = { 
		capital_scope = { continent = europe }
		has_country_flag = cannorian_ardimya_colonist
		OR = {
			culture_is_human = yes
			culture_is_half_elven = yes
		}
		OR = {
			AND = {
				current_age = age_of_reformation
				num_of_owned_provinces_with = {
					value = 30
					region = ardimya_region
					culture_is_human = yes
					is_colony = no
				}
			}
			AND = {
				current_age = age_of_absolutism
				num_of_owned_provinces_with = {
					value = 15
					region = ardimya_region
					culture_is_human = yes
					is_colony = no
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					region = ardimya_region
					is_colony = no
					culture_is_human = yes
				}
				save_event_target_as = ardimyan_rise_province
				
			}
			set_global_flag = ardimya_event_chain_started
		}

	}
	
	option = {
		name = ardimya.1.a
		goto = ardimyan_rise_province
		event_target:ardimyan_rise_province = {
			add_province_modifier = { name = ardimya_rise_of_new_people duration = 18250 }
			change_culture = ardimyan
			hidden_effect = {
				province_event = {
					id = ardimya.3
					days = 7300
				}
				set_variable = {
					which = ardimya_convert_counter
					value = 10
				}
				province_event = {
					id = ardimya.2
					days = 730
				}			
			}
		}
	}
}

	picture = COLONIZATION_eventPicture
	hidden = yes
	
	is_triggered_only = yes
	fire_only_once = yes
	trigger = { 
		always = yes
	}
	
	
	option = {
		name = ardimya.3.a
		every_country = {
			limit = {
				#I think this one is more restricted 
				has_country_flag = cannorian_ardimya_colonist
				capital_scope = { continent = europe }
			}
			every_owned_province = {
				limit = {
					region = ardimya_region
				}
				PREV = {
					change_variable = {
						which = ardimya_provinces
						value = 1
					}				
				}
			}
			if = {
				limit = {  has_saved_event_target = ardimya_overlord  }
				if = {
					limit = {
						check_variable = {
							which = ardimya_provinces
							which = event_target:ardimya_overlord
						}
					}
					save_event_target_as = ardimya_overlord
				}						
			}
			else = {
				save_event_target_as = ardimya_overlord
			}
		}
		
		event_target:ardimya_overlord = {
			country_event = {
				id = ardimya.5
			}
		}
	}
}

			random_owned_province = {
				limit = {
					region = ardimya_region

					NOT = { province_id = event_target:ardimyan_conversion_1 }
				}

				save_event_target_as = ardimyan_conversion_2
			}
			if = {
				limit = { has_saved_event_target = ardimyan_conversion_2 }
				random_owned_province = {
					limit = {
						region = ardimya_region
						is_colony = no
						culture_is_human = yes
						NOT = { culture = ardimyan }
						NOT = { province_id = event_target:ardimyan_conversion_1 }
						NOT = { province_id = event_target:ardimyan_conversion_2 }
					}
					save_event_target_as = ardimyan_conversion_3
				}			
			}		
		}
	}
	option = {
		name = ardimya.4.a

				add_province_modifier = { name = ardimya_rise_of_new_people duration = 18250 }

		}

			limit = { has_saved_event_target = ardimyan_conversion_3 }

				add_province_modifier = { name = ardimya_rise_of_new_people duration = 18250 }
				change_culture = ardimyan
			}
		}		
	}
}

			add_local_autonomy = -60
		}

			duration = -1
		}
		add_country_modifier = {
			name = ardimya_independence_war_income
			duration = 1825
		}
		if = {
			limit = {
				ai = yes
			}
			random_owned_province = {
				limit  = {
					has_port = yes
				}
				build_to_forcelimit = {
					infantry = 0.5
					cavalry = 0.2
					artillery = 0.3
					heavy_ship = 0.2
					light_ship = 0.4
					galley = 0.4
				}
			}
			while = {
				limit = {
					NOT = { manpower_percentage = 0.99 }
				}
				add_manpower = 5
			}			
		}
		else = {
			capital_scope = {
				infantry = ROOT
				infantry = ROOT
				infantry = ROOT
				infantry = ROOT
				infantry = ROOT
			}
			
			random_owned_province = {
				limit = {
					has_port = yes
				}
				heavy_ship = ROOT
				heavy_ship = ROOT
				heavy_ship = ROOT
				galley = ROOT
				galley = ROOT
				galley = ROOT
				galley = ROOT
				galley = ROOT
				galley = ROOT
				galley = ROOT
			}
			add_manpower = 10
		}

		
		declare_war_with_cb = {
			casus_belli = cb_independence_war
			who = overlord
		}
		custom_tooltip = ardimya_independece_timelimit_tt

	}

}
