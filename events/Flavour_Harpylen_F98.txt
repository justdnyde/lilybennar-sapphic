country_event = { #the first gift
	id = flavour_harpylen.156
	title = flavour_harpylen.156.t
	desc = flavour_harpylen.156.d
	picture = ANGRY_MOB_eventPicture
	is_triggered_only = yes
	
	
	immediate = {
		hidden_effect = {
			# The idea is that you gain less magical experience the further away from the target value of harpylen_project_var you are
			# Level 1 target: 5
			# Level 2 target: 12
			# Level 3 target: 9
			# If you hit the target, you gain 200 xp. You gain 25 fewer xp for each point away from the target value you are
			# With 5 events each giving +1, +2, or +3 to harpylen_project_var, the range is 5-15
			# So, final xp = 200 - 25 * abs(target - harpylen_project_var)
			set_country_flag = in_magic_progress_event
			if = {
				limit = { at_least_magic_project_level = { project = harpylen_project level = 2 } }
				change_variable = { harpylen_project_var = -9 }
			}
			else_if = {
				limit = { at_least_magic_project_level = { project = harpylen_project level = 1 } }
				change_variable = { harpylen_project_var = -12 }
			}
			else = { change_variable = { harpylen_project_var = -5 } }

			set_variable = {
				which = harpylen_project_var_distance_from_target
				which = harpylen_project_var
			}

			if = {
				limit = { check_variable = { harpylen_project_var = 0 } }
				multiply_variable = { harpylen_project_var = -1 }
			}

			multiply_variable = { harpylen_project_var = 25 }
			change_variable = { harpylen_project_var = 200 }
		}
	}
	after = { clr_country_flag = in_magic_progress_event }

	# Then give some different loc depending on how close you were
	option = {
		name = flavour_harpylen.156.a
		trigger = {
			NOT = {
				check_variable = {
					which = harpylen_project_var_distance_from_target
					value = -1
				}
			}
		}
		custom_tooltip = jimi_harpy_experience
		hidden_effect = { increase_magic_experience = { unmodified_var_amount = harpylen_project_var } }
	}

	option = {
		name = flavour_harpylen.156.b
		trigger = {
			check_variable = {
				which = harpylen_project_var_distance_from_target
				value = -1
			}
			NOT = {
				check_variable = {
					which = harpylen_project_var_distance_from_target
					value = 2
				}
			}
		}
		custom_tooltip = jimi_harpy_experience
		hidden_effect = { increase_magic_experience = { unmodified_var_amount = harpylen_project_var } }
	}

	option = {
		name = flavour_harpylen.156.c
		trigger = {
			check_variable = {
				which = harpylen_project_var_distance_from_target
				value = 2
			}
		}
		custom_tooltip = jimi_harpy_experience
		hidden_effect = { increase_magic_experience = { unmodified_var_amount = harpylen_project_var } }
	}
}
