
namespace = great_conqueror

country_event = {
	id = great_conqueror.0
	title = great_conqueror.0.t
	desc = great_conqueror.0.d
	picture = ORTHODOX_eventPicture

	trigger = {
		has_dlc = "Rights of Man"
		OR = {
			has_global_flag = great_conqueror
			has_global_flag = mythical_conqueror 
		}
		is_subject = no
		#If delayed conquerors are on, then it can't be the age of discovery.
		OR = {
			NOT = { has_global_flag = delayed_conquerors }
			NOT = { current_age = age_of_discovery }
		}
		ai = yes
		NOT = { has_country_flag = vile_one_vc }
	}
	
	hidden = yes
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = { has_country_flag = previous_ruler_was_great_conqueror_flag }
			clr_country_flag = previous_ruler_was_great_conqueror_flag
		}
		if = {
			limit = { has_country_flag = great_conqueror_flag }
			REB = { subtract_variable = { nbGC = 1 } }
			clr_country_flag = great_conqueror_flag
			set_country_flag = previous_ruler_was_great_conqueror_flag
			export_to_variable = {
				which = ownDev
				value = total_development
				who = ROOT
			}
			add_ruler_modifier = {
				name = legacy_of_great_conqueror
				duration = -1
			}
			every_owned_province = {
				remove_province_modifier = seat_of_centaur_gc
			}
		}
	}
	
	#NO CONQUEROR
	option = {
		name = great_conqueror.0.a
		ai_chance = { 
			factor = 1
			
			modifier = {
				factor = 1.5
				overextension_percentage = 0.5
			}
			
			modifier = {
				factor = 1.5
				average_unrest = 1
			}
			
			modifier = {
				factor = 1.5
				num_of_loans = 5
			}
			
			modifier = {
				factor = 1.5
				corruption = 5
			}
			
			modifier = {
				factor = 1.5
				NOT = { stability = 0 }
			}
			
			modifier = {
				factor = 50
				government = republic
			}
			
			modifier = {
				factor = 25
				government = theocracy
			}
			
			modifier = {
				factor = 10
				is_subject = yes
			}
		}
	}

	#CONQUEROR
	option = {
		name = great_conqueror.0.b
		trigger = {
			NOT = { has_country_flag = previous_ruler_was_great_conqueror_flag } 					#Can't have it 2time in a row
		}
		ai_chance = { 
			factor = 5
			
			modifier = {
				factor = 0
				OR = {
					if = {
						limit = { has_global_flag = mythical_conqueror }
						REB = { check_variable = { nbGC = 13 } }									#No more than 13 Great Conqueror in the world
					}
					else = {
						REB = { check_variable = { nbGC = 10 } }									#No more than 10 Great Conqueror in the world
					}
					is_bankrupt = yes																#Can't be bankrupt bitch
					AND = {
						any_neighbor_country = { ai = no }											#Doesn't target AI next to player before 1550
						NOT = { is_year = 1550 }
					}
					coalition_target = ROOT															#Must not be the coalition target
					NOT = { num_of_cities = 2 }														#Must have more than one province
					AND = {
						government = tribal 														#Only certain tribals allowed
						tribal_elligible_for_gc = no
					}
				}
			}
			
			modifier = {
				factor = 1.5
				is_great_power = yes
			}
			
			modifier = {
				factor = 1.5
				MIL = 3 ADM = 3 DIP = 3
			}
			modifier = {
				factor = 2
				MIL = 4 ADM = 4 DIP = 4
			}
			modifier = {
				factor = 2.5
				MIL = 5 ADM = 5 DIP = 5
			}
			modifier = {
				factor = 3
				MIL = 6 ADM = 6 DIP = 6
			}
			
			modifier = {
				factor = 3
				NOT = { ruler_age = 30 }
			}
			
			modifier = {
				factor = 1.5
				ruler_is_powerful_mage = yes
			}
			
			modifier = {
				factor = 8
				REB = { NOT = { check_variable = { nbGC = 2 } } }									#More chance if less than 2 Great Conqueror
			}
			
			modifier = {
				factor = 4
				REB = { NOT = { check_variable = { nbGC = 4 } } }									#More chance if less than 4 Great Conqueror
			}
			
			modifier = {
				factor = 2
				REB = { NOT = { check_variable = { nbGC = 5 } } }									#More chance if less than 5 Great Conqueror
			}
			
			modifier = {
				factor = 2
				capital_scope = { superregion = escann_superregion }
				total_development = 500
			}
			
			modifier = {																			#Weak neighbours give more chance to become one
				factor = 2
				calc_true_if = {
					all_neighbor_country = {
						NOT = { is_part_of_hre = yes }
						NOT = { alliance_with = ROOT }
						NOT = { is_subject_of = ROOT }
						NOT = { army_strength = { who = ROOT value = 0.4 } }
					}
					amount = 2
				}
			}
			
			modifier = {
				factor = 3
				calc_true_if = {
					all_neighbor_country = {
						NOT = { is_part_of_hre = yes }
						NOT = { alliance_with = ROOT }
						NOT = { is_subject_of = ROOT }
						NOT = { army_strength = { who = ROOT value = 0.4 } }
					}
					amount = 3
				}
			}
			
			modifier = {
				factor = 4
				calc_true_if = {
					all_neighbor_country = {
						NOT = { is_part_of_hre = yes }
						NOT = { alliance_with = ROOT }
						NOT = { is_subject_of = ROOT }
						NOT = { army_strength = { who = ROOT value = 0.4 } }
					}
					amount = 4
				}
			}
			
			modifier = {
				factor = 5
				calc_true_if = {
					all_neighbor_country = {
						NOT = { is_part_of_hre = yes }
						NOT = { alliance_with = ROOT }
						NOT = { is_subject_of = ROOT }
						NOT = { army_strength = { who = ROOT value = 0.4 } }
					}
					amount = 5
				}
			}
			
			modifier = {
				factor = 6
				calc_true_if = {
					all_neighbor_country = {
						NOT = { is_part_of_hre = yes }
						NOT = { alliance_with = ROOT }
						NOT = { is_subject_of = ROOT }
						NOT = { army_strength = { who = ROOT value = 0.4 } }
					}
					amount = 6
				}
			}
			
			modifier = {
				factor = 2
				all_neighbor_country = {
					NOT = { tech_difference = 1 }													#More chance to become one if technologically advanced compared to all neighbor
				} 
			}
			
			modifier = {
				factor = 5
				is_revolution_target = yes
			}
			
			modifier = {
				factor = 3
				power_projection = 50
			}
			
			modifier = {
				factor = 5
				has_global_flag = mythical_conqueror
				is_great_power = yes
			}

			modifier = {
				factor = 3
				centaur_can_become_vc = yes #More common if chance to become Khuraen Caehn
			}
		}
		
		if = {
			limit = {
				has_global_flag = mythical_conqueror
				OR = {
					is_great_power = yes
					total_development = 1000
				}
			}
			create_mythical_conqueror = yes
		}
		else = {
			create_great_conqueror = yes
		}

		#Randomly spawns a variant conqueror
		if = {
			limit = { 
				has_global_flag = variant_conqueror  						# No variant conquerors without turning them on in settings
				any_country = {
					ai = no
					knows_country = ROOT 									# Requires that a player can see the tag
				}
			} 				
			random_list = {
				#Variant Conqueror
				1 = {
					modifier = {
						factor = 5
						ruler_has_personality_or_ancestor = { personality = mythical_conqueror_personality }
					}

					#Vampire Conqueror
					if = {
						limit = {
							has_estate = estate_vampires
						}
						country_event = { id = great_conqueror.3 }
					}

					#Umbral Covenant
					else_if = {
						limit = {
							sarhali_can_become_vc = yes
						}
						set_country_flag = vile_one_vc
						set_global_flag = umbral_covenant_formed
						country_event = { id = great_conqueror.7 }
					}

					#Lich Conqueror
					else = {
						country_event = { id = great_conqueror.1 }
					}
				}
				#No Variant Conqueror
				30 = {
				}
			}
		}
		
	}
}

#Lich-King (AI)
country_event = {
	id = great_conqueror.1
	title = great_conqueror.1.t
	desc = great_conqueror.1.d
	picture = LOOMING_DEATH_eventPicture

	is_triggered_only = yes
	hidden = yes

	trigger = {
		has_not_banned_magic = yes
		NOT = { has_ruler_flag = is_a_homunculus }
	}

	option = {
		name = great_conqueror.1.a
		if = {
			limit = { ruler_is_powerful_mage = yes } #Already has flags
			define_type_to_war_wizard_effect = { type = ruler }
			become_a_witch_king_effect = yes
			become_a_lich_effect = yes
		}
		else = { define_powerful_mage = { type = ruler necromancy = 3 random = 5 lich = yes witch_king = yes war_wizard = yes } }

		while = { #They get some permanent effects, only way to give them undead mil
			limit = { NOT = { is_estate_necromancy_level_3 = yes } }
			magic_school_level_up = { type = country school = necromancy }
		}
		while = { #AI gets free aotd levels, lol
			limit = { NOT = { at_least_magic_project_level = { project = army_of_the_dead level = 3 } } }
			magic_project_level_up = { project = army_of_the_dead }
		}
		change_mana = { amount = 200 }
		
		capital_scope = {
			save_event_target_as = capital_province
		}
		every_country = {
			limit = {
				ai = no
				knows_country = ROOT
			}
			country_event = { id = great_conqueror.2 }
		}
	}
}

country_event = {

	title = great_conqueror.5.t
	desc = great_conqueror.5.d
	picture = CENTAUR_CAMPFIRE_eventPicture

	is_triggered_only = yes
	hidden = yes

	trigger = {
		ai = yes
		centaur_can_become_vc = yes
		has_country_flag = great_conqueror_flag
		total_development = 50
		army_size = 10
		if = {
			limit = { is_at_war = yes }
			war_score = -10									# isn't in the process of getting stomped
			all_war_enemy_countries = { ai = yes }			# isn't fighting a player
		}
	}

	option = {
		name = great_conqueror.5.a
		capital_scope = {
			save_event_target_as = capital_province
		}

		#Upgrade from Great Conqueror to Mythical Conqueror
		create_mythical_conqueror = yes

		form_crimson_caehnate = yes

		every_country = {
			limit = {
				ai = no
				knows_country = ROOT
			}
			country_event = { id = great_conqueror.6 }
		}
	}
}

country_event = {
	id = great_conqueror.7
	title = great_conqueror.7.t
	desc = great_conqueror.7.d
	picture = LOOMING_DEATH_eventPicture

	is_triggered_only = yes
	hidden = yes

	option = {
		name = great_conqueror.7.a
		capital_scope = {
			save_event_target_as = capital_province
		}

		# Can't call the normal events because we need to handle the ruler change logic here.

		form_umbral_covenant = yes

		define_ruler = {
            name = "The Vile One"
            dynasty = " "
            age = 99
            adm = 6
            dip = 1
            mil = 6
            female = yes
        }
        if = {
            limit = {
                NOT = {
                    was_tag = S30
                }
            }
            set_variable = {
                which = S90_concentrate
                value = 0
            }
            set_variable = {
                which = S90_exploit_count
                value = 0
            }
        }
        set_variable = {
            which = S90_festivals
            value = 0
        }
        set_variable = {
            which = S90_scorch_count
            value = 0
        }
        set_variable = {
            which = S90_vileone_battles
            value = 0
        }
        set_variable = {
            which = S90_major_rift_count
            value = 0
        }
        set_country_flag = count_battles_won
        set_country_flag = count_wars_won
        create_mythical_conqueror = yes
        add_ruler_personality = conqueror_personality
        add_ruler_personality = shadow_planetouched_personality
		define_powerful_mage = {
			type = ruler
			enchantment = 3
			evocation = 2
			illusion = 2
			necromancy = 1
		}
        make_ruler_immortal = yes
        add_ruler_modifier = {
        	name = umbral_covenant_gc
        	duration = 9125
        }
        every_owned_province = {
            limit = {
                region = shadow_swamp_region
            }
            cede_province = S70
        }
        every_ally = {
        	limit = {
        		NOT = { religion = shadow_pact }
        		NOT = { religion = night_coven }
        	}
        	break_alliance = ROOT
        	break_marriage = ROOT
        }
        S70 = {
            country_event = {
                id = flavor_umbral_covenant.2
            }
        }
        country_event = {
            id = flavor_umbral_covenant.999
        }

        change_religion = shadow_pact
        every_owned_province = {
        	change_religion = shadow_pact
        }

        create_unit_forcelimit_percentage = {
        	type = infantry
        	special_unit_category = mamluks
        	force_limit_percentage = 0.6
        }

        custom_tooltip = umbral_alliance_tt

		every_country = {
			limit = {
				ai = no
				knows_country = ROOT
			}
			country_event = { id = great_conqueror.8 }
		}
	}
}
