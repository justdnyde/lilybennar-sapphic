country_event = {
	id = flavour_RSP.10
	title = flavour_RSP.10.t
	desc = flavour_RSP.10.d
	picture = COASTLINE_MAP_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		ai = no
		has_country_flag = RSP_treasure_hunt_system_enabled
		OR = {
			is_or_was_tag = { tag = J95 }
			is_or_was_tag = { tag = J96 }
			is_or_was_tag = { tag = J97 }
			is_or_was_tag = { tag = J98 }
		}
		any_province = {
			has_province_flag = RSP_buried_treasure
			NOT = { country_or_subject_holds = ROOT }
		}
	}

	immediate = {
		hidden_effect = {
			random_province = {
				limit = { 
					has_province_flag = RSP_buried_treasure
					NOT = { country_or_subject_holds = ROOT }
				}
				save_event_target_as = treasuremapprovince
			}
		}
	}

	option = {
		name = flavour_RSP.10.a
		goto = treasuremapprovince
		event_target:treasuremapprovince = {
			clr_province_flag = RSP_buried_treasure
			add_claim = ROOT
			add_province_modifier = {
				name = RSP_x_marks_the_spot
				duration = 3650
			}
		}
		custom_tooltip = flavour_RSP_10_a
	}
}

country_event = {
	id = flavour_RSP.11
	title = flavour_RSP.11.t
	desc = flavour_RSP.11.d
	picture = LAKE_TREASURE_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = flavour_RSP.11.a
		custom_tooltip = flavour_RSP_11_a
		hidden_effect = { 
			trigger_switch = {
				on_trigger = government_rank
				3 = { random_list = { 25 = { add_treasury = 600 } 40 = { add_treasury = 450 add_prestige = 75 } 25 = { add_treasury = 300 add_adm_power = 90 add_mil_power = 90 add_dip_power = 90} 10 = { add_prestige = 75 add_adm_power = 90 add_mil_power = 90 add_dip_power = 90} } }
				2 = { random_list = { 25 = { add_treasury = 400 } 40 = { add_treasury = 300 add_prestige = 50 } 25 = { add_treasury = 200 add_adm_power = 60 add_mil_power = 60 add_dip_power = 60} 10 = { add_prestige = 50 add_adm_power = 60 add_mil_power = 60 add_dip_power = 60} } }
				1 = { random_list = { 25 = { add_treasury = 200 } 40 = { add_treasury = 150 add_prestige = 25 } 25 = { add_treasury = 100 add_adm_power = 30 add_mil_power = 30 add_dip_power = 30} 10 = { add_prestige = 25 add_adm_power = 30 add_mil_power = 30 add_dip_power = 30} } }
			}
			if = {
				limit = { has_country_modifier = RSP_improved_buried_treasures }
				trigger_switch = {
					on_trigger = tag
					J95 = { 
						random_known_country = { 
							limit = { 
								NOT = { is_enemy = J95 }
								capital_scope = { continent = europe }
							}
							add_opinion = { who = J95 modifier = RSP_rhasail_treasure_opinion }
						}
					}
					J96 = {
						random_province = {
    						limit = {
								has_port = yes
								superregion = ruin_proper_superregion
								NOT = { religion = uelos_cult }
							}
    						change_religion = uelos_cult
						}
					}
					J97 = { increase_magic_experience = { amount = 50 } }
					J98 = { artifice_unlock_random_invention_effect = yes }
				}
			}
			change_variable = {
				which = RSP_buried_treasures_found
				value = 1
			}
		}
	}
