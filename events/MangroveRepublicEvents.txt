	id = mangrove.3
	title = mangrove.3.t
	desc = mangrove.3.d
	picture = COLONIZATION_eventPicture
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			export_to_variable = {
				which = mangrove_manpower
				value = manpower
			}
			export_to_variable = {
				which = mangrove_money
				value = treasury
			}
			every_known_country = {
				limit = { has_government_attribute = mangrove_republic }
				if = {
					limit = { has_saved_event_target = mangrove_highest_other }
					if = {
						limit = {
							check_variable = {
								which = mangrove_influence
								which = event_target:mangrove_highest_other
							}
						}
						save_event_target_as = mangrove_highest_other
					}
				}
				else = {
					save_event_target_as = mangrove_highest_other
				}
			}
		}
	}

	option = {
		name = mangrove.3.a
		trigger = {
			anb_require_all_power = { VAL = 50 }

			factor = 1
		}

			mangrove_donate = yes
		}
		custom_tooltip = mangrove_donate_tt
	}

		name = mangrove.3.b

				which = mangrove_manpower
				which = mangrove_manpower_check

		}

		}

		NOT = { has_global_flag = delete_insyaa_region }
	}

		name = none

		mangrove_new_target_province = yes

			limit = { has_saved_global_event_target = mangrove_target_province }
			event_target:mangrove_leader = {
				event_target:mangrove_target_province = {
					change_trade_goods = wool
					add_core = PREV
					cede_province = PREV
					add_permanent_province_modifier = {
						name = mangrove_reclamation_site
						duration = 1850 # ~5 years
					}
					add_local_autonomy = -50
					set_variable = { mangrove_donations = 0 }

					# decide trade goods
					province_event = {
						id = mangrove.1020
						days = 730 # 2 years
					}
					province_event = {
						id = mangrove.1021
						days = 1460 # 4 years
					}
					province_event = {
						id = mangrove.104
						days = 1825 # 5 years
					}
				}
			}
			mangrove_event = { COUNTRY_EVENT = "mangrove.103" }

# vying handler
country_event = {
	id = mangrove.108
	title = none
	desc = none
	picture = none
	is_triggered_only = yes
	hidden = yes

	option = {
		name = none

		clr_global_flag = mangrove_voting

		# find largest tag and player tag
		every_country = {
			limit = { has_government_attribute = mangrove_republic }
			# the largest tag determines the type of vying
			if = {
				limit = {
					has_saved_event_target = mangrove_largest_tag
				}
				if = {
					limit = { total_development = event_target:mangrove_largest_tag }
					save_event_target_as = mangrove_largest_tag
				}
			}
			else = {
				save_event_target_as = mangrove_largest_tag
			}
			# also find largest player tag
			if = {
				limit = { ai = no }
				if = {
					limit = {
						has_saved_event_target = mangrove_largest_player
					}
					if = {
						limit = { total_development = event_target:mangrove_largest_player }
						save_event_target_as = mangrove_largest_player
					}
				}
				else = {
					save_event_target_as = mangrove_largest_player
				}
			}
		}

		# choose path based on largest tag
		if = {
			limit = {
				event_target:mangrove_largest_tag = {
					OR = {
						primary_culture = togavaopaolo
						tag = P22 # Anatu'agao
					}
				}
			}
			# choose diplo path
			set_global_flag = mangrove_dip_vying
		}
		else_if = {
			limit = {
				event_target:mangrove_largest_tag = {
					OR = {
						primary_culture = aliiula
						tag = P21 # Seila Agaga
					}
				}
			}
			# choose mil path
			set_global_flag = mangrove_mil_vying
		}
		else = {
			# choose random path
			random_list = {
				50 = {
					set_global_flag = mangrove_dip_vying
				}
				50 = {
					set_global_flag = mangrove_mil_vying
				}
			}
		}

		if = { # player can change type of vying
			limit = { has_saved_event_target = mangrove_largest_player }
			event_target:mangrove_largest_player = {
				country_event = { id = mangrove.109 }
			}
		}
		else = { # otherwise just begin the vying
			country_event = {
				id = mangrove.110
			}
		}
	}
}

			if = {

				if = {
					limit = {
						event_target:mangrove_most_punia_logi = {
							NOT = {
								check_variable = {
									which = mangrove_punia_logi
									which = PREV
								}
							}
						}
					}
					save_event_target_as = mangrove_most_punia_logi
				}
			}
			else = {
				save_event_target_as = mangrove_most_punia_logi
			}
		}
		# check for tie
		if = {
			limit = {
				event_target:mangrove_most_punia_logi = {
					any_known_country = {
						has_government_attribute = mangrove_republic
						check_variable = {
							which = mangrove_punia_logi
							which = PREV
						}
					}
				}
			}
			if = {
				limit = { NOT = { has_global_flag = mangrove_dip_vying_tied } }
				# check for winner in 2 years
				set_global_flag = mangrove_dip_vying_tied
				country_event = {
					id = mangrove.116
					days = 730
				}
				mangrove_event = { COUNTRY_EVENT = "mangrove.117" }
			}
			else = {
				# random winner among the tied
				clr_global_flag = mangrove_dip_vying_tied
				event_target:mangrove_most_punia_logi = {
					random_country = {
						limit = {
							has_government_attribute = mangrove_republic
							check_variable = {
								which = mangrove_punia_logi
								which = PREV
							}
						}
						country_event = { id = mangrove.118 }
					}
				}
			}
		}
		else = {
			clr_global_flag = mangrove_dip_vying_tied
			event_target:mangrove_most_punia_logi = {
				country_event = { id = mangrove.118 }
			}
		}
	}
