country_event = {
	id = flavor_nansalen.1
	title = flavor_nansalen.1.t
	desc = flavor_nansalen.1.d
	
	picture = COURT_eventPicture
	
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			F33 = {
				if = { limit = { ai = yes }
					if = { limit = { NOT = { religion = bulwari_sun_cult } }
						change_religion = bulwari_sun_cult
						add_prestige = 100
					}
					set_country_flag = supernaturally_bound_faith # To prevent them from flipping to Jadd.
					add_country_modifier = { name = F80_cmod_akalses_temp_buff duration = 18250 }
				}
				clr_country_flag = F80_cflag_will_aid_nansalen
			}
			F42 = { clr_country_flag = F80_cflag_will_aid_nansalen }
		}
	}

	option = {
		name = "flavor_nansalen.1.a"
		ai_chance = {
			factor = 1
		}
		if = { limit = { incident_variable_value = { incident = incident_F80_kings_queens_kneeling value = 1 } }
			increase_nsc_isolation_level = yes
			add_country_modifier = { name = F80_cmod_kings_queens_increase_isolation duration = 9125 }
		}
		else = {
			decrease_nsc_isolation_level = yes
			add_country_modifier = { name = F80_cmod_kings_queens_decrease_isolation duration = 9125 }
		}

		if = { limit = { F33 = { ai = no } }
			harpy_roost_se_allow_construction_of_extra_roost = yes
		}
		else = {
			custom_tooltip = F80_tt_akalses_special_subject
			hidden_effect = {
				create_subject = {
					subject_type = F80_F33_loyal_subject
					subject = F33
				}
			}
		}

		if = { limit = { F33 = { ai = yes } harpy_hills_region = { OR = { country_or_subject_holds = F42 country_or_subject_holds = F33 } } }
			custom_tooltip = F80_tt_harpy_hills_ceded
			hidden_effect = {
				harpy_hills_region = { limit = { OR = { country_or_subject_holds = F42 country_or_subject_holds = F33 } }
					owner = { remove_core = PREV }
					cede_province = ROOT
				}
			}
		}

		add_war_exhaustion = -4

		custom_tooltip = F80_tt_irregular_demonsterization
		if = { limit = { NOT = { adm_tech = 7 } }
			set_country_flag = no_demonsterization_events_till_tech_7
		}
		hidden_effect = { start_demonsterization_effect = yes }

		end_incident = incident_F80_kings_queens_kneeling
		clr_country_flag = active_nsc_incident_flag
		swap_non_generic_missions = yes
	}
}

# Peering Past the Pass effect
country_event = {
	id = flavor_nansalen.100
	title = flavor_nansalen.100.t
	desc = flavor_nansalen.100.d
	
	picture = COURT_eventPicture
	
	hidden = yes
	is_triggered_only = yes

	after = {
		clr_country_flag = F80_cflag_divination_discover_forbidden_plains_province
		clr_country_flag = F80_cflag_divination_discover_forbidden_plains_area
		clr_country_flag = F80_cflag_divination_discover_forbidden_plains_region
	}

	option = {
		name = "flavor_nansalen.100.a"
		ai_chance = {
			factor = 1
		}

		# Sea Zones, need to be handled a bit differently since they don't count for neighbor_province.
		if = { limit = { OR = { has_discovered = 1441 has_discovered = 1443 has_discovered = 1666 has_discovered = 5309 } } 
			5312 = { discover_country = ROOT }
		}

		if = { limit = { OR = { has_discovered = 1379 has_discovered = 1380 has_discovered = 1384 has_discovered = 1386 has_discovered = 5317 } } 
			1444 = { discover_country = ROOT }
		}

		forbidden_lands_superregion = { 
			limit = { 
				has_discovered = ROOT
				NOT = { has_province_flag = F80_pflag_has_spread_discovery } 
			}
			set_province_flag = F80_pflag_has_spread_discovery
			every_neighbor_province = {
				limit = {
					superregion = forbidden_lands_superregion
					NOT = { has_discovered = ROOT }
				}
				set_province_flag = F80_pflag_divination_discovery 
			}
			every_empty_neighbor_province = {
				limit = {
					superregion = forbidden_lands_superregion
					NOT = { has_discovered = ROOT }
				}
				set_province_flag = F80_pflag_divination_discovery
			}
			sea_zone = { discover_country = ROOT }
		}
		if = { limit = { has_country_flag = F80_cflag_divination_discover_forbidden_plains_province }
			forbidden_lands_superregion = { 
				limit = { 
					has_province_flag = F80_pflag_divination_discovery
					NOT = { has_discovered = ROOT }
				}
				discover_country = ROOT
				clr_province_flag = F80_pflag_divination_discovery
			}
		}
		else_if = { limit = { has_country_flag = F80_cflag_divination_discover_forbidden_plains_area }
			forbidden_lands_superregion = { 
				limit = { 
					has_province_flag = F80_pflag_divination_discovery
					NOT = { has_discovered = ROOT }
				}
				area = {
					discover_country = ROOT
					clr_province_flag = F80_pflag_divination_discovery
				}
			}
		}
		else_if = { limit = { has_country_flag = F80_cflag_divination_discover_forbidden_plains_region }
			forbidden_lands_superregion = { 
				limit = { 
					has_province_flag = F80_pflag_divination_discovery
					NOT = { has_discovered = ROOT }
				}
				region = {
					discover_country = ROOT
					clr_province_flag = F80_pflag_divination_discovery
				}
			}
		}

		# Sea zones again, for area or region.
		if = { 
			limit = { 
				OR = {
					has_country_flag = F80_cflag_divination_discover_forbidden_plains_area
					has_country_flag = F80_cflag_divination_discover_forbidden_plains_region
				}
			}
			if = { limit = { OR = { has_discovered = 1441 has_discovered = 1443 has_discovered = 1666 has_discovered = 5309 } } 
				5312 = { discover_country = ROOT }
			}
	
			if = { limit = { OR = { has_discovered = 1379 has_discovered = 1380 has_discovered = 1384 has_discovered = 1386 has_discovered = 5317 } } 
				1444 = { discover_country = ROOT }
			}
		}
		# Check whether everything has been discovered.
		if = { 
			limit = {
				forbidden_lands_superregion = {
					type = all
					has_discovered = ROOT
				}
			}
			forbidden_lands_superregion = {
				clr_province_flag = F80_pflag_has_spread_discovery
			}
			set_country_flag = F80_cflag_discovered_everything_in_forbidden_lands
			# Always discover the trade node sea zones in this case, just to be 100% sure.
			5312 = { discover_country = ROOT }
			1444 = { discover_country = ROOT }
		}
	}
}
