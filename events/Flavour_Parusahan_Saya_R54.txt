# A Desperate Attempt
country_event = {
	id = flavor_parusahan_saya.3
	title = flavor_parusahan_saya.3.t
	desc = flavor_parusahan_saya.3.d
	picture = EXCALIBUR_eventPicture

	hidden = no
	is_triggered_only = yes

	immediate = {
		# allies leave the raj and join you against the vizier (will rejoin raj after)
		# must not call anyone into the war
		hidden_effect = {
			every_ally = {
				limit = {
					has_government_attribute = is_raja_subject
				}
				if = {
					limit = { is_subject_of_type = great_daimyo_vassal }
					set_country_flag = R54_senapti_flag
					4411 = {
						change_variable = {
							which = rajNbSenapti
							value = -1
						}
					}
				}
				else_if = {
					limit = { is_subject_of_type = daimyo_vassal }
					set_country_flag = R54_prabhi_flag
				}
				
				grant_independence = yes
				join_all_offensive_wars_of = R54
			}
			# raja joins war against you
			overlord = {
				join_all_defensive_wars_of = event_target:raj_vizier_global_target
			}
		}
	}

	option = {

			}
		}

	id = flavor_parusahan_saya.8

	desc = flavor_parusahan_saya.8.d
	picture = DIVERSE_COURT_eventPicture

	hidden = no
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			if = {
				limit = {
					calc_true_if = {
						amount = 3
						has_country_modifier = R54_bypassed_tollkeepers
						has_country_modifier = R54_artisanal_obstruction
						has_country_modifier = R54_landless_loomlords
					}
				}
				set_country_flag = R54_baron_revolt_severity_high
			} 
			else_if = {
				limit = {
					calc_true_if = {
						amount = 2
						has_country_modifier = R54_bypassed_tollkeepers
						has_country_modifier = R54_artisanal_obstruction
						has_country_modifier = R54_landless_loomlords
					}
				}
				set_country_flag = R54_baron_revolt_severity_med
			}
			else = {
				set_country_flag = R54_baron_revolt_severity_low
			}

			if = {
				limit = { has_country_flag = R54_baron_revolt_severity_high }
				set_variable = {
					which = R54_baron_revolt_rebel_count
					value = 15
				}
			}
			else_if = {
				limit = { has_country_flag = R54_baron_revolt_severity_med }
				set_variable = {
					which = R54_baron_revolt_rebel_count
					value = 10
				}
			}

				limit = { has_country_flag = R54_baron_revolt_severity_low }
				set_variable = {
					which = R54_baron_revolt_rebel_count
					value = 5
				}
			}
		}

			set_country_flag = R54_barons_revolted

				every_owned_province = {
					add_local_autonomy = 30
				}
			}
			
			add_estate_loyalty = {
				estate = estate_uppercastes
				loyalty = -40
			}
			add_estate_loyalty = {
				estate = estate_middlecastes
				loyalty = -40
			}
			add_estate_loyalty = {
				estate = estate_lowercastes
				loyalty = -40
			}
			custom_tooltip = flavor_parusahan_saya.8_tt4
		}
		else_if = {
			limit = {
				has_country_flag = R54_baron_revolt_severity_med
			}
			add_stability = -2
			custom_tooltip = flavor_parusahan_saya.8_tt2
			hidden_effect = {
				every_owned_province = {
					add_local_autonomy = 20
				}
			}
			add_estate_loyalty = {
				estate = estate_uppercastes
				loyalty = -30
			}
			add_estate_loyalty = {
				estate = estate_middlecastes
				loyalty = -30
			}
			add_estate_loyalty = {
				estate = estate_lowercastes
				loyalty = -30
			}
			custom_tooltip = flavor_parusahan_saya.8_tt4
		}

			add_stability = -1
