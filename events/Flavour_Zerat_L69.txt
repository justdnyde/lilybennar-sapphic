country_event = {
	id = flavour_zerat.1000
	title = flavour_zerat.1000.t
	desc = flavour_zerat.1000.d
	picture = DIPLOMACY_eventPicture

	is_triggered_only = yes
	hidden = yes

	trigger = {
		tag = L69 #Zerat
		has_country_flag = anzara_mechanic_unlocked
	}

	immediate = {
		if = {
			limit = {
				NOT = {
					estate_loyalty = {
						estate = estate_nobles
						loyalty = 30
					}
				}
			}
			export_estate_influence_to_var = {
				estate = estate_nobles
				var = zerat_noble_influence
			}
			multiply_variable = { #Every Point Of Noble Influence Reduces 0.01 Gov Power
				which = zerat_noble_influence
				value = 0.01
			}
			set_variable = { #Used For TT
				which = zerat_resurgence_influence_lost
				which = zerat_noble_influence
			}
			gov_power_var_effect_granular = { #Reduce Gov Power
				mechanic = anzara_resurgence_mechanic
				power = anzara_power
				var = zerat_noble_influence
				sign = -
			}
		}
		else = {
			set_variable = { #Used For TT
				which = zerat_resurgence_influence_lost
				value = 0
			}
		}
		if = {
			limit = { has_country_flag = anzara_hegemony }
			every_subject_country = {
				limit = {
					is_subject_of_type = anzara_march
				}
				if = {
					limit = { liberty_desire = 50 }
					ROOT = {
						change_variable = {
							which = anzara_march_disloyalty
							value = 1
						}
					}
				}
				else = {
					ROOT = {
						change_variable = {
							which = anzara_march_loyalty
							value = 1
						}
					}
				}
			}
			#+0.05 Per Loyal Anzara March
			multiply_variable = {
				which = anzara_march_loyalty
				value = 0.05
			}
			set_variable = { #For TT Purposes
				which = anzara_march_loyalty_gained
				which = anzara_march_loyalty
			}
			gov_power_var_effect_granular = { #Gain Gov Power
				mechanic = anzara_resurgence_mechanic
				power = anzara_power
				var = anzara_march_loyalty
				sign = +
			}

			#-0.1 Per Disloyal Anzara March
			multiply_variable = {
				which = anzara_march_disloyalty
				value = 0.1
			}
			set_variable = { #For TT Purposes
				which = anzara_march_disloyalty_lost
				which = anzara_march_disloyalty
			}
			gov_power_var_effect_granular = { #Gain Gov Power
				mechanic = anzara_resurgence_mechanic
				power = anzara_power
				var = anzara_march_disloyalty
				sign = -
			}

			subtract_variable = { #Sum The Lost/Gain Amount For TT
				which = anzara_march_loyalty_gained
				which = anzara_march_disloyalty_lost
			}
		}
	}

	option = {
		name = flavour_zerat.1.a
	}
}
