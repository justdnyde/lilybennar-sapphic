namespace = landslide_infantry
#
# Landslide Hazard (& Synergy) Pulse Events
#
# Reasoning: 
# Landslide Infantry is inherently risky and dangerous.
# Both to the user, their allies and the general surroundings. 
# Having them is a boon, but should come at a cost you can slowly help mitigate.

country_event = {
	id = landslide_infantry.0
	title = "landslide_infantry.0.title"
	desc = "landslide_infantry.0.desc"
	hidden = yes
	picture = WOUNDED_SOLDIERS_eventPicture

	trigger = {
		has_government_attribute = landslide_infantry_events
		any_owned_province = { has_province_modifier = landslide_infantry_hazards_possible }
		num_of_tercio = 1
	}
	is_triggered_only = yes

	option = {
		name = none
		
		random_list = {
			12 = { # 50% - 40% Chance (Depending on .20, .21 & .22 choices)
				# No Event happens
				# Chance increased for each additional trigger unmet (Impossible for more than 6 to be met at once)
			}
			3 = { # [Hazard] Landslide Practice damages Town
				trigger = {
					NOT = { has_country_flag = landslide_synergy_town }
				}
				country_event = { id = landslide_infantry.1 }
			}
			6 = { # [Synergy] Landslide Practice defends Town
				trigger = {
					has_country_flag = landslide_synergy_town
				}
				country_event = { id = landslide_infantry.2 }
			}
			3 = { # [Hazard] Landslide Infantry Brawl
				trigger = {
					NOT = { has_country_flag = landslide_synergy_brawl }
					NOT = { has_country_modifier = yanacotle_shamed_landslide_infantry }
				}
				country_event = { id = landslide_infantry.3 }
			}
			6 = { # [Synergy] Bonds of Landslide Infantry Comradery
				trigger = {
					has_country_flag = landslide_synergy_town
					NOT = { has_country_modifier = yanacotle_shamed_landslide_infantry }
					NOT = { has_country_modifier = yanacotle_high_spirited_landslide_infantry }
					NOT = { has_country_modifier = yanacotle_regimented_landslide_infantry }
				}
				country_event = { id = landslide_infantry.4 }
			}
			2 = { # [Hazard] Landslide Infantry starts Fire
				trigger = {
					NOT = { has_country_flag = landslide_synergy_fire }
					any_owned_province = { 
						has_province_modifier = landslide_infantry_hazards_possible
						OR = { has_terrain = woods has_terrain = jungle }
					}
				}
				country_event = { id = landslide_infantry.5 }
			}
			4 = { # [Synergy] Landslide Infantry help contain fire
				trigger = {
					has_country_flag = landslide_synergy_fire
					NOT = { has_country_modifier = yanacotle_honoured_landslide }
					any_owned_province = { 
						has_province_modifier = landslide_infantry_hazards_possible
						OR = { has_terrain = woods has_terrain = jungle }
					}
				}
				country_event = { id = landslide_infantry.6 }
			}
			2 = { # [Hazard] Landslide Infantry damages Fort
				trigger = {
					NOT = { has_country_flag = landslide_synergy_fort }
					any_owned_province = { 
						has_province_modifier = landslide_infantry_hazards_possible
						fort_level = 1
						controlled_by = PREV
					}
				}
				country_event = { id = landslide_infantry.7 }
			}
			4 = { # [Synergy] Landslide Infantry fortify Fort
				trigger = {
					is_at_war = yes
					has_country_flag = landslide_synergy_fort
					any_owned_province = { 
						has_province_modifier = landslide_infantry_hazards_possible
						fort_level = 1
						controlled_by = PREV
					}
					NOT = { has_country_modifier = yanacotle_landslide_siege }
				}
				country_event = { id = landslide_infantry.8 }
			}
			1 = { # [Hazard] Landslide Infantry triggers Landslide
				trigger = {
					NOT = { has_country_flag = landslide_synergy_landslide }
					any_owned_province = { 
						has_province_modifier = landslide_infantry_hazards_possible			
						OR = { has_terrain = hills has_terrain = mountain }
					}
				}
				country_event = { id = landslide_infantry.9 }
			}
			2 = { # [Synergy] Landslide Infantry ward of Eruption
				trigger = {
					has_government_attribute = landslide_infantry_events
					has_country_flag = landslide_synergy_landslide 
					any_owned_province = { 
						has_province_modifier = landslide_infantry_hazards_possible
						province_group = insyaa_volcanoes 
					}			
					NOT = { has_country_modifier = yanacotle_inspired_by_landslide_heroes }
				}
				country_event = { id = landslide_infantry.10 }
			}
			1 = { # [Hazard] Landslide Infantry burns Barracks
				trigger = {
					NOT = { has_country_flag = landslide_synergy_barracks }
				}
				country_event = { id = landslide_infantry.11 }
			}
			2 = { # [Synergy] Landslide Infantry aid in Rebuilding
				trigger = {
					has_country_flag = landslide_synergy_barracks 
					any_owned_province = { 
						has_province_modifier = landslide_infantry_hazards_possible
						devastation = 15 
					}
					is_at_war = no
				}
				country_event = { id = landslide_infantry.12 }
			}
		}
	}
}

	desc = "landslide_infantry.1.desc"

	picture = WOUNDED_SOLDIERS_eventPicture
	
	trigger = {}
	is_triggered_only = yes
	immediate = {
		hidden_effect = { 
				random_owned_province = { 
				limit = { has_province_modifier = landslide_infantry_hazards_possible }

			}
		}
	}

	#Option 1 - Pay for the recruits funeral
	option = {
		name = "landslide_infantry.1.a"

		name = "landslide_infantry.1.c"

		event_target:landslide_hazard = { 
			add_devastation = 10 
			random = { 
				chance = 50 
				add_random_development = -1
			}
		}
	}
}
