#religious action for Collect Children of Fire
country_event = {
	id = yenbenmirra.3
	title = yenbenmirra.3.t
	desc = yenbenmirra.3.d
	picture = MECHANIM_AWAKENING_eventPicture
	
	is_triggered_only = yes
	major = no
	
	option = {
		name = yenbenmirra.3.a
		custom_tooltip = collect_children_of_fire_tt
		trigger = {
			OR = {
				any_tribal_land = { has_province_modifier = dormant_mechanim }
				any_owned_province = { has_province_modifier = dormant_mechanim } 
			}
		}
		hidden_effect = {
			oceania = {
				type = random
				amount = 1
				limit = {
					has_province_modifier = dormant_mechanim
					OR = {
						tribal_land_of = ROOT
						owned_by = ROOT
					}
				}
				ROOT = {
					change_variable = {
						which = children_of_fire_count
						value = 200
					}
				}
				remove_claim = P10
				remove_claim = P11
				remove_claim = P12
				remove_claim = P13
				remove_claim = P14
				remove_claim = P15
				remove_claim = P16
				remove_claim = P17
				remove_claim = P19
				remove_claim = P20
				remove_province_modifier = dormant_mechanim
				add_permanent_province_modifier = {
					name = scoured_dormant_mechanim
					duration = -1
				}
			}
			yenbenmirra_update_cof_modifier = yes
			yenbenmirra_update_hinui_hunters = { target_scope = capital_scope }
		}
		ai_chance = { factor = 100 }
	}
	option = {
		name = yenbenmirra.3.b
		add_church_power = 100
		ai_chance = { factor = 1 }
	}
}

#startup event for Yenbenmirra tribes to set their children of fire collection number
country_event = {
	id = yenbenmirra.4
	title = yenbenmirra.4.t
	desc = yenbenmirra.4.d
	picture = COMET_SIGHTED_eventPicture
	
	is_triggered_only = yes
	major = no

	trigger = {
		has_reform = yenbenmirra_tribe
		NOT = {
			is_year = 1445
		}
	}
	option = {
		name = yenbenmirra.4.a
		custom_tooltip = yenbenmirra_startup_tt
		add_country_modifier = {
			name = children_of_fire_collection_small
			duration = -1
		}
		hidden_effect = {
			set_variable = {
				which = children_of_fire_count
				value = 0
			}
			every_known_country = {
				limit = {
					OR = {
						culture_is_kobold = yes
						culture_group = tagaala
					}
					historical_friend_with = ROOT
				}
				set_ai_attitude = {
					attitude = attitude_friendly
					who = ROOT
				}
			}
			every_known_country = {
				limit = {
					OR = {
						culture_group = yenbenmirra
					}
					NOT = { historical_friend_with = ROOT }
				}
				set_ai_attitude = {
					attitude = attitude_hostile
					who = ROOT
				}
			}
			insyaan_lowlands_superregion = {
				limit = {
					has_province_modifier = dormant_mechanim
				}
				add_permanent_claim = ROOT
			}
		}
	}
}
