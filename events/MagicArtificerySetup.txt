#Rise of Artificery / Choose Mode
country_event = { 
	id = magic_artificery_setup.1
	title = magic_artificery_setup.1.t
	desc = magic_artificery_setup.1.d
	picture = INVENTION_eventPicture

	is_triggered_only = yes


	

	#mage only
	option = {
		name = magic_artificery_setup.1.a
		trigger = {
			artificery_locked_mixed_mode = no
			artificery_locked_artificery_mode = no
			artificery_tag_needs_artificers = no
		}
		artifice_magic_switch_to_magic_only_mode = yes
	}
	#mage/artificery
	option = {
		name = magic_artificery_setup.1.b
		trigger = {
			artificery_locked_mage_mode = no
			artificery_locked_artificery_mode = no
		}
		artifice_magic_switch_to_mixed_mode = yes
		hidden_effect = {
			country_event = {
				id = magic_artificery_setup.2
				days = 1
			}
		}
	}
	#artificers only
	option = {
		name = magic_artificery_setup.1.c
		trigger = {
			artificery_locked_mage_mode = no
			artificery_locked_mixed_mode = no
			artificery_tag_needs_mages = no
		}
		artifice_magic_switch_to_artificery_only_mode = yes
		hidden_effect = {
			country_event = {
				id = magic_artificery_setup.2
				days = 1
			}
			country_event = {
				id = magic_artificery_setup.3
				days = 1
			}
		}
	}

	#special mage only
	option = {
		name = magic_artificery_setup.1.g
		highlight = yes
		trigger = {
			has_country_flag = unlock_mage_only
			NOT = { has_government_attribute = is_technocracy }
		}
		custom_tooltip = unlocked_mode_tt
		artifice_magic_switch_to_magic_only_mode = yes
	}
	#special mage/artificery
	option = {
		name = magic_artificery_setup.1.e
		highlight = yes
		trigger = {
			hidden_trigger = {
				has_country_flag = unlock_mixed_mode
			}
		}
		custom_tooltip = unlocked_mode_tt
		artifice_magic_switch_to_mixed_mode = yes
		hidden_effect = {
			country_event = {
				id = magic_artificery_setup.2
				days = 1
			}
		}
	}
	#special artificers only
	option = {
		name = magic_artificery_setup.1.f
		highlight = yes
		trigger = {
			has_country_flag = unlock_artificery_only
			NOT = { has_government_attribute = is_magocracy }
			ruler_is_not_lich = yes #no liches with artificery
			ruler_is_not_witch_king = yes #no witch-kings with artificery_only
		}
		custom_tooltip = unlocked_mode_tt
		artifice_magic_switch_to_artificery_only_mode = yes
		hidden_effect = {
			country_event = {
				id = magic_artificery_setup.2
				days = 1
			}
			country_event = {
				id = magic_artificery_setup.3
				days = 1
			}
		}
	}
	#failstate mage only
	option = {
		name = magic_artificery_setup.1.h
		highlight = yes
		trigger = {
			OR = {
				artificery_locked_mixed_mode = yes
				artificery_locked_artificery_mode = yes
				artificery_tag_needs_artificers = yes
			}
			artificery_locked_mage_mode = yes
			hidden_trigger = {
				NOT = { has_country_flag = unlock_mixed_mode }
			}
		}
		custom_tooltip = failstate_mode_tt
		artifice_magic_switch_to_magic_only_mode = yes
	}
}

country_event = { 
	id = magic_artificery_setup.3
	title = magic_artificery_setup.3.t
	desc = magic_artificery_setup.3.d
	picture = INVENTION_eventPicture

	is_triggered_only = yes
	hidden = yes

	trigger = { has_not_banned_magic = yes } #you're banning them through artificery
	
	immediate = {
		
	}

	option = {
		name = magic_artificery_setup.3.a
		if = {
			limit = { is_year = 1446 }
			add_country_modifier = { 
				name = expelled_mages 
				duration = 365
			}
			if = {
				limit = {
					exists = A85
					has_estate_privilege = estate_mages_organization_magisterium
				}
				A85 = {
					add_opinion = {
						who = ROOT
						modifier = expelled_the_magisterium
						years = 20
					}
				}
			}
		}
		
		#gets rid of all mages advisors and stuffs in your country
		if = {
			limit = {
				advisor = court_mage
			}
			kill_advisor = court_mage
		}
		country_event = { id = magic_system.99 } #Abort current magic advancement
		close_all_magic_menus = yes #Keeping it open was bugging the game

country_event = { #Easter Egg event where you convert away from an artificery culture
	id = magic_artificery_setup.5
	title = magic_artificery_setup.5.t
	desc = magic_artificery_setup.5.d
	picture = OVEREXTENSION_eventPicture

	is_triggered_only = yes
	trigger = {
		has_country_flag = artificery_unlocked #You have artificers
		NOT = { has_country_flag = non_artificery_culture } #You didn't already get this event

		NOT = { #You don't have artificery through MT content
			artificery_locked_artificery_mode = yes
			artificery_locked_mixed_mode = yes
			artificery_mt_locked_mode = yes
			artificery_tag_needs_artificers = yes
		}

		NOT = { has_institution = manufactories } #You did NOT unlock artificery through having manufactories
		#Technically Bulwari religion nations have a thing here but it's locked by manufactories too

		NOT = { is_artificery_culture_government = yes } #You are the WRONG culture to be having artificery
	}

	immediate = { set_country_flag = non_artificery_culture }

	option = {
		name = magic_artificery_setup.5.a
		custom_tooltip = artificery_halved_until_manufactories_tt
	}
}
