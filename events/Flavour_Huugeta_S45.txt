# Poison Their Hearts, fired from declarewar diplomatic action
country_event = {
	id = huugeta.2 
	title = huugeta.2.t
	desc = huugeta.2.d
	picture = KING_ASSASSINATION_eventPicture
	is_triggered_only = yes

	option = {
		name = huugeta.2.a
		trigger = {
			event_target:s45_poison_their_hearts_target = { 
				OR = {
					NOT = { ruler_has_max_personalities = yes }
					ruler_has_personality = inspiring_leader_personality
				}
				NOT = { ruler_has_personality = craven_personality }
			}
		}

		event_target:s45_poison_their_hearts_target = {
			if = {
				limit = {
					ruler_has_personality = inspiring_leader_personality
				}
				remove_ruler_personality = inspiring_leader_personality
			}

			add_ruler_personality = craven_personality
		}


		add_spy_network_in = {
			who = event_target:s45_poison_their_hearts_target
			value = -30
		}
	}

	option = {
		name = huugeta.2.b

		trigger = {
			event_target:s45_poison_their_hearts_target = {
				has_leaders = {
					value = 1
					include_monarch = no
					include_heir = no
					type = general
				}
			}
		}

		event_target:s45_poison_their_hearts_target = {
			kill_leader = {
				type = general
			}
		}

		add_spy_network_in = {
			who = event_target:s45_poison_their_hearts_target
			value = -30
		}
	}

	option = {
		name = huugeta.2.c
		
		event_target:s45_poison_their_hearts_target = {

				name = S45_poisoned_hearts

			}
		}

			who = event_target:s45_poison_their_hearts_target
			value = -30
		}
	}

	option = {
		name = huugeta.2.e
		# Choose to remain hidden
