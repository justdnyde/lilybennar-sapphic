country_event = {
	id = flavour_samsumbat.12
	title = flavour_samsumbat.12.title
	desc = flavour_samsumbat.12.desc
	picture = SONGHAI_RULER_eventPicture

	is_triggered_only = yes

	# Choose A if trust of Samsumbat is 80+ or relative army strength is < 50%
	option = {
		name = flavour_samsumbat.12.a

		trigger = { # Using a trigger instead of ai_chance with factors of 0 since sometimes the AI will just ignore the 0 weights
			OR = {
				ai = no
				trust = {
					who = FROM
					value = 80
				}
				NOT = {
					army_strength = {
						who = FROM
						value = 0.5
					}
				}
			}
		}

		FROM = {
			create_subject = {
				subject_type = tributary_state
				subject = PREV
			}
		}
	}

	# Otherwise, choose B if opinion of Samsumbat is 160+
	option = {
		name = flavour_samsumbat.12.b

		trigger = { # Using a trigger instead of ai_chance with factors of 0 since sometimes the AI will just ignore the 0 weights
			OR = {
				ai = no
				AND = {
					NOT = {
						trust = {
							who = FROM
							value = 80
						}
					}
					army_strength = {
						who = FROM
						value = 0.5
					}
					has_opinion = {
						who = FROM
						value = 160
					}
				}
			}
		}

		change_religion = nirakhetist
		if = {
			limit = {
				NOT = { government = republic }
			}
			change_government = republic
		}
		add_government_reform = peasants_republic
	}

	# Otherwise, choose C
	option = {
		name = flavour_samsumbat.12.c
		
		trigger = { # Using a trigger instead of ai_chance with factors of 0 since sometimes the AI will just ignore the 0 weights
			OR = {
				ai = no
				AND = {
					NOT = {
						trust = {
							who = FROM
							value = 80
						}
					}
					army_strength = {
						who = FROM
						value = 0.5
					}
					NOT = {
						has_opinion = {
							who = FROM
							value = 160
						}
					}
				}
			}
		}

		break_alliance = FROM
		reverse_add_casus_belli = {
			target = FROM # Samsumbat
			type = cb_force_satrapy
			months = 60 # 5 years
		}
	}
}
