namespace = marmorac

country_event = { # handler
	id = marmorac.0
	title = marmorac.0.t
	desc = marmorac.0.d
	picture = wihgfx_FEMALE_RULER_eventPicture

	is_triggered_only = yes
	hidden = yes

	trigger = {
		owns = 2023
		2023 = {
			has_great_project = { type = marmorac_carnival tier = 2 }
			can_use_marmorac_carnival = yes
		}
	}

	option = {
		name = marmorac.0.a
		random_list = {
			1 = {
				trigger = {
					NOT = { has_country_flag = marmorac_1_no_repeat }
				}
				country_event = {
					id = marmorac.1
				}
			}
			1 = {
				trigger = {
					NOT = { has_country_flag = marmorac_2_no_repeat }
				}
				country_event = {
					id = marmorac.2
				}
			}
			1 = {
				trigger = {
					NOT = { has_country_flag = marmorac_3_no_repeat }
					has_regency = no
				}
				country_event = {
					id = marmorac.3
				}
			}
			1 = {
				trigger = {
					NOT = { has_country_flag = marmorac_4_no_repeat }
					NOT = { religious_unity = 1 }
				}
				country_event = {
					id = marmorac.4
				}
			}
		}
	}
}


country_event = { # A Lovely Evening
	id = marmorac.1
	title = marmorac.1.t
	desc = marmorac.1.d
	picture = wihgfx_FEMALE_RULER_eventPicture

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			clr_country_flag = marmorac_2_no_repeat
			clr_country_flag = marmorac_3_no_repeat
			clr_country_flag = marmorac_4_no_repeat
			
			set_country_flag = marmorac_1_no_repeat
		}
	}

	option = { # T2 safe
		name = marmorac.1.a
		
		trigger = {
			2023 = {
				NOT = { has_great_project = { type = marmorac_carnival tier = 3 } }
			}
		}

		add_stability_or_adm_power = yes
	}
	option = { # T2 gamble
		name = marmorac.1.b
		trigger = {
			2023 = {
				NOT = { has_great_project = { type = marmorac_carnival tier = 3 } }
			}
		}

		random_list = {
			50 = {
				add_2_stability_or_adm_power = yes
			}
			50 = {
				add_stability = -2
			}
		}
	}

	option = { # T3 safe
		name = marmorac.1.a
		trigger = {
			2023 = {
				has_great_project = { type = marmorac_carnival tier = 3 }
			}
		}
		if = { # 1 stab + 50 admin, like so to combine the numbers
			limit = { stability = 3 }
			add_adm_power = 100
		}
		else = {
			add_stability = 1
			add_adm_power = 50
		}
	}
	option = { # T3 gamble
		name = marmorac.1.b
		trigger = {
			2023 = {
				has_great_project = { type = marmorac_carnival tier = 3 }
			}
		}
		random_list = {
			66 = {
				if = { # 2 stab + 50 admin, like so to combine the numbers
					limit = { stability = 3 }
					add_adm_power = 150
				}
				else_if = {
					limit = { stability = 2 }
					add_stability = 1
					add_adm_power = 100
				}
				else = {
					add_stability = 2
					add_adm_power = 50
				}
			}
			33 = {
				add_stability = -1
			}
		}
	}
}
