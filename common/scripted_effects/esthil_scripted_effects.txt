# Country scope
B54_se_calc_monthly_change = {
	set_variable = { which = var_B54_gov_power_monthly_change value = 0 }
	if = { limit = { has_country_flag = B54_cflag_esthil_necronomics_death_tax_upgrade }
		export_to_variable = { which = var_B54_dev value = trigger_value:total_development }
		change_variable = { which = var_B54_gov_power_monthly_change which = var_B54_dev }
	}
	if = { limit = { NOT = { has_country_flag = B54_cflag_deadmines_disabled } }
		subtract_variable = { which = var_B54_gov_power_monthly_change which = var_B54_deadmines_upkeep }
	}
	if = { limit = { NOT = { has_country_flag = B54_cflag_reapfields_disabled } }
		subtract_variable = { which = var_B54_gov_power_monthly_change which = var_B54_reapfields_upkeep }
	}
}

# Country scope
B54_se_upgrade_deadmines_reapfields = {
	custom_tooltip = B54_tt_necronomics_upgrade_$type$
	hidden_effect = {
		if = { limit = { has_country_flag = B54_cflag_$type$_level_2 }
			set_country_flag = B54_cflag_$type$_level_3
			clr_country_flag = B54_cflag_$type$_level_2
			set_variable = { which = var_B54_$type$_cost value = @necronomics_level_3_cost }

			if = { limit = { has_country_flag = B54_cflag_$type$_disabled }
				every_owned_province = { limit = { has_province_modifier = B54_pmod_$type$_inactive_2 }
					remove_province_modifier = B54_pmod_$type$_inactive_2
					add_province_modifier = { name = B54_pmod_$type$_inactive_3 duration = -1 }
				}
			}
			else = {
				every_owned_province = { limit = { has_province_modifier = B54_pmod_$type$_active_2 }
					remove_province_modifier = B54_pmod_$type$_active_2
					add_province_modifier = { name = B54_pmod_$type$_active_3 duration = -1 }
				}
			}
		}
		else_if = { limit = { has_country_flag = B54_cflag_$type$_level_1 }
			set_country_flag = B54_cflag_$type$_level_2
			clr_country_flag = B54_cflag_$type$_level_1
			set_variable = { which = var_B54_$type$_cost value = @necronomics_level_2_cost }

			if = { limit = { has_country_flag = B54_cflag_$type$_disabled }
				every_owned_province = { limit = { has_province_modifier = B54_pmod_$type$_inactive_1 }
					remove_province_modifier = B54_pmod_$type$_inactive_1
					add_province_modifier = { name = B54_pmod_$type$_inactive_2 duration = -1 }
				}
			}
			else = {
				every_owned_province = { limit = { has_province_modifier = B54_pmod_$type$_active_1 }
					remove_province_modifier = B54_pmod_$type$_active_1
					add_province_modifier = { name = B54_pmod_$type$_active_2 duration = -1 }
				}
			}
		}
		else = {
			set_country_flag = B54_cflag_$type$_level_1
			set_variable = { which = var_B54_$type$_cost value = @necronomics_level_1_cost }
		}

		set_variable = { which = var_B54_$type$_upkeep which = var_B54_$type$_num }
		multiply_variable = { which = var_B54_$type$_upkeep which = var_B54_$type$_cost }
	}
}

B54_se_subtract_development_cost_effect = {
	hidden_effect = {
		# Determine whether other sources of dev cost reductions are enough to reach the floor dev cost value, if so, scale the gov power cost accordingly.
		owner = {
			export_to_variable = {
				which = var_B54_local_dev_cost
				value = modifier:local_development_cost
				who = PREV
			}
			# These calculations happen after local dev has been increased, meaning that local cost is slightly higher than it "should" be, which must be offset.
			PREV = {
				if = { limit = { development = 150 }
					owner = { subtract_variable = { which = var_B54_local_dev_cost value = 0.45 } }
				}
				else_if = { limit = { development = 140 }
					owner = { subtract_variable = { which = var_B54_local_dev_cost value = 0.42 } }
				}
				else_if = { limit = { development = 130 }
					owner = { subtract_variable = { which = var_B54_local_dev_cost value = 0.39 } }
				}
				else_if = { limit = { development = 120 }
					owner = { subtract_variable = { which = var_B54_local_dev_cost value = 0.36 } }
				}
				else_if = { limit = { development = 110 }
					owner = { subtract_variable = { which = var_B54_local_dev_cost value = 0.33 } }
				}
				else_if = { limit = { development = 100 }
					owner = { subtract_variable = { which = var_B54_local_dev_cost value = 0.30 } }
				}
				else_if = { limit = { development = 90 }
					owner = { subtract_variable = { which = var_B54_local_dev_cost value = 0.27 } }
				}
				else_if = { limit = { development = 80 }
					owner = { subtract_variable = { which = var_B54_local_dev_cost value = 0.24 } }
				}
				else_if = { limit = { development = 70 }
					owner = { subtract_variable = { which = var_B54_local_dev_cost value = 0.21 } }
				}
				else_if = { limit = { development = 60 }
					owner = { subtract_variable = { which = var_B54_local_dev_cost value = 0.18 } }
				}
				else_if = { limit = { development = 50 }
					owner = { subtract_variable = { which = var_B54_local_dev_cost value = 0.15 } }
				}
				else_if = { limit = { development = 40 }
					owner = { subtract_variable = { which = var_B54_local_dev_cost value = 0.12 } }
				}
				else_if = { limit = { development = 30 }
					owner = { subtract_variable = { which = var_B54_local_dev_cost value = 0.09 } }
				}
				else_if = { limit = { development = 20 }
					owner = { subtract_variable = { which = var_B54_local_dev_cost value = 0.06 } }
				}
				else_if = { limit = { development = 10 }
					owner = { subtract_variable = { which = var_B54_local_dev_cost value = 0.03 } }
				}
			}
			export_to_variable = {
				which = var_B54_country_dev_cost
				value = modifier:development_cost
			}
			export_to_variable = {
				which = var_B54_country_apc_reduction
				value = modifier:all_power_cost
			}
			set_variable = { which = var_B54_cumulative_dev_discount value = 0 }
			change_variable = { which = var_B54_cumulative_dev_discount which = var_B54_local_dev_cost }
			change_variable = { which = var_B54_cumulative_dev_discount which = var_B54_country_dev_cost }
			change_variable = { which = var_B54_cumulative_dev_discount which = var_B54_country_apc_reduction }

			# Special case: capital gets up to -50% dev cost discount based on total development, which is not counted as local dev cost for some reason.
			if = { limit = { PREV = { is_capital = yes } }
				# It did not work until I did it like this.
				if = { limit = { total_development = 1000 } # Max discount.
					set_variable = { which = var_B54_capital_total_dev_discount value = 0.5 }
				}
				if = { limit = { NOT = { total_development = 1000 } }
					export_to_variable = { which = var_B54_capital_total_dev_discount value = total_development }
					divide_variable = { which = var_B54_capital_total_dev_discount value = 2000 }
				}
				subtract_variable = { which = var_B54_cumulative_dev_discount which = var_B54_capital_total_dev_discount }
			}

			# Add 0.9 and flip sign of var_B54_cumulative_dev_discount to get the amount of "wasted" dev cost reduction
			set_variable = { which = var_B54_dev_discount_wasted which = var_B54_cumulative_dev_discount }
			change_variable = { which = var_B54_dev_discount_wasted value = 0.9 }
			multiply_variable = { which = var_B54_dev_discount_wasted value = -100 } # Go from 0.0X to X, makes it work better.
			if = { limit = { check_variable = { which = var_B54_dev_discount_wasted value = 0 } }
				# There is wasted dev cost reduction.
				# Figure out how much dev cost reduction if from the gov mechanic.
				if = { limit = { PREV = { has_province_modifier = B54_pmod_necronomics_aid_development_3 } }
					set_variable = { which = var_B54_gov_power_dev_discount value = 20 }
				}
				if = { limit = { PREV = { has_province_modifier = B54_pmod_necronomics_aid_development_2 } }
					set_variable = { which = var_B54_gov_power_dev_discount value = 15 }
				}
				if = { limit = { PREV = { has_province_modifier = B54_pmod_necronomics_aid_development_1 } }
					set_variable = { which = var_B54_gov_power_dev_discount value = 10 }
				}

				if = { limit = { check_variable = { which = var_B54_dev_discount_wasted which = var_B54_gov_power_dev_discount } }
					# Gov power cost reduction was superfluous, no cost.
					set_variable = { which = var_B54_gov_power_cost value = 0 }
				}
				if = { limit = { NOT = { check_variable = { which = var_B54_dev_discount_wasted which = var_B54_gov_power_dev_discount } } }
					# Between 0% and 100% of the gov power discount actually made a difference.
					# Also 100x'ing some variables here, otherwise it seems to round down.
					set_variable = { which = var_B54_gov_power_share_wasted which = var_B54_dev_discount_wasted }
					divide_variable = { which = var_B54_gov_power_share_wasted which = var_B54_gov_power_dev_discount }
					multiply_variable = { which = var_B54_gov_power_share_wasted value = 100 }
					set_variable = { which = var_B54_gov_power_cost value = 100 }
					subtract_variable = { which = var_B54_gov_power_cost which = var_B54_gov_power_share_wasted }
					multiply_variable = { which = var_B54_gov_power_cost value = 25 }
				}
			}
			if = { limit = { NOT = { check_variable = { which = var_B54_dev_discount_wasted value = 0 } } }
				# No wasted dev cost reduction, full cost.
				set_variable = { which = var_B54_gov_power_cost value = 2500 }
			}
			subtract_variable = { which = var_B54_total_gov_power which = var_B54_gov_power_cost }
			custom_tooltip = B54_tt_aid_development_cost
			if = { limit = { NOT = { check_variable = { which = var_B54_total_gov_power value = 2500 } } }
				B54_se_disable_aid_development = yes
				country_event = { id = flavor_esthil.106 }
			}
		}	
	}
}

		change_variable = { which = raise_undead which = land_units_lost }

		round_variable = { which = raise_undead value = -1 } #truncate decimal portion

		set_variable = { which = var_B54_raise_gov_power which = raise_undead }

		# Check whether player has Undead Mil.
		if = { limit = { has_country_modifier = undead_military }
			# Multiply by fraction assigned to raising undead.
			multiply_variable = { which = var_B54_raise_gov_power which = var_B54_raise_fraction_gov_power }
			change_variable = { which = var_B54_total_gov_power which = var_B54_raise_gov_power }
			B54_se_calc_thousands = yes

			multiply_variable = { which = raise_undead which = var_B54_raise_fraction_manpower }
			
			if = { limit = { NOT = { has_country_flag = hide_necromancy_on_battle } }
				country_event = { id = magic_project_army_of_the_dead.21 }
			}
			else = {
				divide_variable = { raise_undead = 1000 }
				var_effect_granular = { effect = add_manpower var = raise_undead sign = + }				
			}
		}
		else_if = { 
			limit = {
				OR = {
					has_country_flag = B54_cflag_unrestricted_toiling_souls
					all_war_enemy_countries = { 
						OR = { 
							has_country_modifier = orcish_administration 
							has_country_modifier = goblin_administration 
							AND = {
								is_subject = yes
								overlord = {
									OR = { 
										has_country_modifier = orcish_administration 
										has_country_modifier = goblin_administration
									}
								}
							}
						} 
					} 
				}
			}
			if = { limit = { is_estate_necromancy_level_3 = yes } 
				multiply_variable = { which = var_B54_raise_gov_power value = 50 }
			} 
			else_if = { limit = { is_estate_necromancy_level_2 = yes } 
				multiply_variable = { which = var_B54_raise_gov_power value = 40 }
			}
			else_if = { limit = { is_estate_necromancy_level_1 = yes }
				multiply_variable = { which = var_B54_raise_gov_power value = 30 }
			}
			else = {
				multiply_variable = { which = var_B54_raise_gov_power value = 20 }
			}

			change_variable = { which = var_B54_total_gov_power which = var_B54_raise_gov_power }
