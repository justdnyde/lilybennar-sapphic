				}
			}
			manago_poku_discover_pillar_dragon = yes
		}
		else_if = {
			limit = {
				capital_scope = {
					OR = {
						superregion = insyaan_lowlands_superregion
						superregion = charkuchar_superregion
						region = titanoflora_valley_region
						region = titanoflora_savanna_region

				}
			}
			manago_poku_discover_pillar_sandworm = yes
		}
		else = {
			manago_poku_discover_pillar_forest = yes
		}
	}
}

		limit = {

manago_poku_migrate_effects = {
	if = {
		limit = { owner = { religion = manago_poku has_country_flag = manago_poku_not_settled } }
		# Intro event
		owner = {
			if = {
				limit = {
					NOT = {
						has_country_flag = manago_poku_setup
					}
				}
				set_country_flag = manago_poku_setup
				country_event = {
					id = manago_poku.56
				}
			}
		}
		# Church power on migration
		owner = { add_church_power = 10 } 
		
		# Boosted migration from T2 reform
		if = {
			limit = { owner = { has_reform = mechanim_learning } }
			owner = { add_church_power = 5 }
		}

		# Church power from dormant mechanim
		if = {
			limit = { has_province_modifier = dormant_mechanim NOT = { has_province_modifier = scoured_dormant_mechanim } }
			if = {
				limit = { owner = { ai = no } }
				remove_province_modifier = dormant_mechanim
				add_permanent_province_modifier = {
					name = scoured_dormant_mechanim
					duration = -1
				}
			}
			owner = {
				add_church_power = 75
			}
		}

		# First visit to a Holohana
		if = {
			limit = {
				owner = { NOT = { has_country_flag = manago_poku_visited_holohana } }
				has_province_modifier = holohana_rumbling
			}
			owner = {
				set_country_flag = manago_poku_visited_holohana
				country_event = {
					id = manago_poku.57
				}
			}
		}
		else_if = {
			limit = {
				owner = { NOT = { has_country_flag = manago_poku_visited_rumbling } }

					any_neighbor_province = { has_province_modifier = holohana_rumbling }

			}
		}
	}

		remove_country_modifier = mechanim_automatory_modifier_02
	}
	else = {
		remove_country_modifier = mechanim_automatory_modifier_01
	}
}

# Gives catch-up buffs to an AI mechanim tag ($tag$) that has just settled or spawned.
# Also boosts the settlement province ($target_scope$) and colonizes empty area provinces around it.
mechanim_ai_settle_buffs = {
	$target_scope$ = {
		add_base_tax = 3
		add_base_production = 3
		add_base_manpower = 2
		area = {
			limit = {
				is_empty = yes
			}
			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
			change_culture = awoken_mechanim
			change_religion = manago_poku
			add_core = $tag$
			cede_province = $tag$
		}
	}
	$tag$ = {
		every_owned_province = {
			add_institution_embracement = { which = feudalism value = 100 }
			add_institution_embracement = { which = renaissance value = 100 }
		}
		every_owned_province = {
			limit = {
				superregion = hinuilands_superregion
				NOT = { province_group = gene_pillars }
			}
			add_province_modifier = { name = mechanim_ward_conduit duration = -1 }
		}
		while = {
			limit = {
				NOT = { adm_tech = 8 } # Usually spawn in 1519 -- 1444+50+25
			}
			add_adm_tech = 1
		}
		while = {
			limit = {
				NOT = { dip_tech = 8 }
			}
			add_dip_tech = 1
		}
		while = {
			limit = {
				NOT = { mil_tech = 8 }
			}
			add_mil_tech = 1
		}
		add_country_modifier = { name = manago_poku_exploring_ai duration = 14600 }
	}
}

# Spawns an AI mechanim formable at a target scope ($target_scope$) as $tag$
# Then gives buffs via the above SE
spawn_ai_mechanim_at_province = {
	$target_scope$ = {
		change_culture = awoken_mechanim
		change_religion = manago_poku
		add_core = $tag$
		cede_province = $tag$
	}
	$tag$ = {
		add_country_modifier = { name = centralization_modifier duration = 7300 }
		set_country_flag = formed_end_game_tag_flag
	}
	mechanim_ai_settle_buffs = { tag = $tag$ target_scope = $target_scope$ }
}

# Kills off a random AI wandering mechanim tribe in $region$
# Limited to not target subjects or tags bordering the player or tags at war
mechanim_depopulate_tribe = {

				region = $region$
			}

				OR = {
					is_subject = yes
					any_neighbor_country = { ai = no }

				}
			}
		}
		every_owned_province = {
			cede_province = ---
		}
	}
}

# Teleports the Mechanim tribe back to a specific gene pillar province.
mechanim_teleport_to_pillar_scripted_effect = {
	if = {
		limit = {
			$pillar_province$ = {
				OR = {
					is_empty = yes
					AND = {
						has_empty_adjacent_province = yes
						any_empty_neighbor_province = { NOT = { has_province_flag = holohana_is_here_flag } }
					}
				}
			}
		}
		if = { 
			limit = { $pillar_province$ = { is_empty = yes } }
			$pillar_province$ = { save_event_target_as = teleport_province }
		}
		else = {
			$pillar_province$ = {
				random_empty_neighbor_province = { 
					limit = { 
						NOT = { has_province_flag = holohana_is_here_flag }
					}
					save_event_target_as = teleport_province
				}
			}
		}
		capital_scope = {
			event_target:teleport_province = {
				add_core = ROOT
				cede_province = ROOT
				change_religion = ROOT
				change_culture = ROOT
				PREV = {
					while = {
						limit = {
							infantry_in_province = ROOT
						}
						kill_units = {
							who = owner
							type = infantry
							amount = 1
						}
						PREV = {
							infantry = ROOT
						}
					}
					copy_building_to_target_province = { building = workshop target = PREV gold = yes }
					copy_building_to_target_province = { building = temple target = PREV }
					copy_building_to_target_province = { building = marketplace target = PREV }
					copy_building_to_target_province = { building = barracks target = PREV }
					copy_building_to_target_province = { building = shipyard target = PREV }
					copy_building_to_target_province = { building = courthouse target = PREV }
					copy_building_to_target_province = { building = dock target = PREV }
					copy_building_to_target_province = { building = regimental_camp target = PREV }
					remove_core = ROOT
					destroy_province = yes
				}
			}
		}
	}
	else = { custom_tooltip = occupied_cant_teleport_tt }
}
