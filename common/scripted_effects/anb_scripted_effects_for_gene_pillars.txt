	hidden_effect = {
		owner = {
			if = {
				limit = { NOT = { has_country_flag = aPUI_gene_pillar_interaction_setup_done } }

				set_aPUI_var = { name = gene_pillars_party_level value = 1 }
				
				set_aPUI_var = { name = gene_pillar_xp value = 0 }

				aPUI_gene_pillar_setup_country_classes = yes

				gene_pillars = {
					limit = { owned_by = PREV }
					aPUI_gene_pillar_setup_province = yes

			} 
		}
	}

aPUI_gene_pillar_reset_province = {
	hidden_effect = {
		set_province_flag = aPUI_gene_pillar_interaction_setup_done_province
	
		set_aPUI_var = { name = gene_pillar_health value = 3 }
		set_aPUI_cap = { name = gene_pillar_health amount = 10 }
		set_aPUI_var = { name = gene_pillar_health_max value = 3 }
		set_aPUI_cap = { name = gene_pillar_health_max amount = 10 }
		
		set_aPUI_var = { name = gene_pillar_xp which = owner } # if we haven't set up, this is 0, otherwise it's our current xp - necessary for e.g. on_province_owner_change conquering new gene pillars while having XP
		# we also need to copy the capacity from the owner, but we must set it to a safe value if the owner doesn't have it set up and not 0, to avoid divide-by-zero errors
		if = {
			limit = { owner = { aPUI_check_capacity_variable = { name = gene_pillar_xp value = 1 } } }
			set_aPUI_cap = { name = gene_pillar_xp which = owner }
		}
		else = {
			set_aPUI_cap = { name = gene_pillar_xp amount = 1 }
		} # failsafe... we shouldn't really need it if all goes well - TODO: determine if necessary
		set_aPUI_var = { name = gene_pillar_loot which = owner } # Loot is stored between expeditions
		set_aPUI_var = { name = gene_pillar_progress value = 0 }

		set_variable = { which = aPUI_gene_pillar_loot_capacity value = 100 }
		set_variable = { which = aPUI_gene_pillar_progress_capacity value = 10 }

		update_aPUI_var_bar = { name = gene_pillar_xp }
		update_aPUI_var_bar = { name = gene_pillar_progress }
		
		# set_aPUI_var = { name = gene_pillars_party_level which = owner }
		[[new_owner]
			gene_pillars_setup_clr_province_class_flags_rogue = yes
			gene_pillars_setup_clr_province_class_flags_ranger = yes
			gene_pillars_setup_clr_province_class_flags_fighter = yes
			gene_pillars_setup_clr_province_class_flags_cleric = yes
			gene_pillars_setup_clr_province_class_flags_mage = yes
			gene_pillars_setup_clr_province_class_flags_artificer = yes
			gene_pillar_UI_setup_party_slots = yes
		]
		aPUI_gene_pillars_setup_update_UI = yes
	}
}
aPUI_gene_pillar_setup_province = {
	if = {
		limit = {
			NOT = { has_province_flag = aPUI_gene_pillar_interaction_setup_done_province }
		}
		aPUI_gene_pillar_reset_province = { new_owner = yes }
	}
}

aPUI_gene_pillar_update_all_gene_pillar_progress_bars = {
	# the XP bar and Loot bar in each gene pillar window is mirroring a country variable,
	# so when you are not in an expedition you can see the bar in all expedition windows and it should have the same value in each
	# thus, when finishing or abandoning an expedition, we need to update the XP + Loot in all owned gene pillars
	# as doing so allows you to see the other gene pillar windows again
	gene_pillars = {
		limit = { owned_by = PREV }
		set_aPUI_var = { name = gene_pillar_xp which = owner }
		set_aPUI_cap = { name = gene_pillar_xp which = owner } # necessary in case a level-up increased these
		update_aPUI_var_bar = { name = gene_pillar_xp }
		set_aPUI_var = { name = gene_pillar_loot which = owner }
		update_aPUI_var_bar = { name = gene_pillar_loot }
	}
}

update_aPUI_health_bar = {
	if = { limit = { check_variable = { which = aPUI_$name$_var value = 0 } }
		set_variable = { which = aPUI_$name$_framevar which = aPUI_$name$_var }
		divide_variable = { which = aPUI_$name$_framevar value = 10 } #hardcoded max
		multiply_variable = { which = aPUI_$name$_framevar value = 100 }
	}
	else = { set_variable = { which = aPUI_$name$_framevar value = 0 } }
}

# Ends the expedition (unsuccessful)
# the actual effect is in the immediate of the two country_events -
# this is because this scripted effect is used indirectly in hundreds of places (f.e. everywhere that does the lose hp effect)
# and that would mean hundreds of copies of it in memory - like this, only the event call is copied in memory
# NOTE: this does mean that this will not work in a scripted effect else_if/else/trigger_switch as it would not fire the event,
# but it would be broken anyway as it would be unable to fire gene_pillars.30 in that case
gene_pillar_abandon_expedition_effect = {
	hidden_effect = {
		[[province_notification_event]	country_event = { id = gene_pillars.3 } ]
		[[country_notification_event]	country_event = { id = gene_pillars.4 } ]
	}
}
gene_pillar_abandon_expedition_effect_helper = {
	gene_pillars_expedition_finished_roll_final_injuries_effect = yes
	gene_pillars_injury_clear_UI_slots_and_province_flags = yes

	hidden_effect = {
		gene_pillar_clear_all_class_country_flags = yes
		aPUI_clear_all_country_exploration_flags = yes
		aPUI_clear_all_country_exploration_modifiers = yes

	}
	hidden_effect = {
		# Reset progress to 0
		event_target:gene_pillar_exploration_province = {
			aPUI_clear_all_province_exploration_flags = yes
			aPUI_clear_all_province_exploration_modifiers = yes
			aPUI_gene_pillar_reset_province = yes
		}

		# TODO: hacky - we have to clear the XP and Loot here as you can see the expedition window again if you hold the finishing events open
		# but we also need to update XP again in each completion event, because you might level up and change your XP again
		# and you may have maxed your Loot, thereby resetting it
		# improve this
		aPUI_gene_pillar_update_all_gene_pillar_progress_bars = yes

		#Pillar ends in failure
		[[province] event_target:gene_pillar_exploration_province = { province_event = { id = gene_pillars.31 } } ]
		[[country] country_event = { id = gene_pillars.32 } ]
	}
}

			limit = { gene_pillars_has_class_in_party = { class = mage_B } }
			custom_tooltip = gene_pillars_add_loot_effect_mage_B_bonus_tt

		}
		if = {
			limit = { gene_pillars_has_class_in_party = { class = artificer_A } }
			custom_tooltip = gene_pillars_add_loot_effect_artificer_A_bonus_tt
			hidden_effect = { change_variable = { which = temp value = 0.25 } }

		hidden_effect = {
			multiply_variable = { which = temp value = $amount$ }
			change_aPUI_var = { name = gene_pillar_loot which = temp } # Update country level so we save between expedition

				set_variable = { which = temp which = PREV }

	]

	#Change on province, refresh Gene Pillar UI
	hidden_effect = {
		change_aPUI_var = { name = gene_pillar_loot value = $amount$ } # Update country level so we save between expedition

				}

gene_pillars_level_up_check_helper = {
	hidden_effect = {
		if = {
			limit = {
				#Don't have a specific scripted trigger
				owner = { aPUI_check_variable = { name = gene_pillar_xp which = aPUI_gene_pillar_xp_capacity } }
			}
			#Nor specific effect

			owner = {
				subtract_variable = { which = aPUI_gene_pillar_xp_var which = aPUI_gene_pillar_xp_capacity }
				change_variable = { which = aPUI_gene_pillars_party_level_var value = 1 }
				if = { # Went from level 1 to level 2
					limit = { gene_pillars_is_exactly_party_level = { level = 2 } }
					gene_pillars = {
						limit = { owned_by = PREV } #Change visual as you unlock slot 2
						set_variable = { which = aPUI_gene_pillar_party_slot_framevar_2 value = @empty }
					}
					set_aPUI_cap = { name = gene_pillar_xp amount = 100 }
				}
				else_if = { # Went from level 2 to level 3
					limit = { gene_pillars_is_exactly_party_level = { level = 3 } }
					gene_pillars = {
						limit = { owned_by = PREV } #Change visual as you unlock slot 3
						set_variable = { which = aPUI_gene_pillar_party_slot_framevar_3 value = @empty }
					}
					set_aPUI_cap = { name = gene_pillar_xp amount = 100 }
	
					if = {
						limit = { NOT = { has_country_modifier = mechanim_administration } }
						set_country_flag = gene_pillars_unlocked_fighter_B
						set_country_flag = gene_pillars_unlocked_fighter_C
					}
				}
				else_if = { # Went from level 3 to level 4
					limit = { gene_pillars_is_exactly_party_level = { level = 4 } }
					gene_pillars = {
						limit = { owned_by = PREV } #Change visual as you unlock slot 4
						set_variable = { which = aPUI_gene_pillar_party_slot_framevar_4 value = @empty }
					}
					set_aPUI_cap = { name = gene_pillar_xp amount = 200 }
	
					if = {
						limit = { NOT = { has_country_modifier = mechanim_administration } }
						set_country_flag = gene_pillars_unlocked_cleric_B
					}
				}
				else_if = { # Went from level 4 to level 5
					limit = { gene_pillars_is_exactly_party_level = { level = 5 } }
					gene_pillars = {
						limit = { owned_by = PREV } #Change visual as you unlock slot 5
						set_variable = { which = aPUI_gene_pillar_party_slot_framevar_5 value = @empty }
					}
					set_variable = { which = aPUI_gene_pillar_xp_var value = 200 }
				}
			}
			#Then get the updated values
			set_aPUI_var = { name = gene_pillar_xp which = owner }
			set_aPUI_cap = { name = gene_pillar_xp which = owner }
			# Necessary as this is called without change/add/lose XP effect
			update_aPUI_var_bar = { name = gene_pillar_xp }
			# Level up notification
			province_event = { id = gene_pillars.6 }
		}
	}
}

# This fires in the immediate of the reward event to calculate the loot, and then
# the event option will actually give it to you
# If you have 100 loot points, make sure we only select each reward once
gene_pillars_randomise_loot_event = {
	owner = {
		random_list = {
			10 = {
				trigger = {
					NOT = {
						has_country_flag = gene_pillars_has_mage_A
						has_country_flag = gene_pillars_has_artificer_A
						has_country_flag = gene_pillars_loot_good_ruler
					}
				}
				set_country_flag = gene_pillars_loot_good_ruler
			}
			10 = {
				trigger = {
					NOT = {
						has_country_flag = gene_pillars_has_mage_A
						has_country_flag = gene_pillars_has_artificer_A
						has_country_flag = gene_pillars_loot_institution
					}
				}
				set_country_flag = gene_pillars_loot_institution
			}
			10 = { # Get precursor relics
				trigger = {
					NOT = {
						has_country_flag = gene_pillars_has_artificer_A
						has_country_flag = gene_pillars_loot_precursor_relics
						has_country_flag = gene_pillars_loot_damestear
					}
					NOT = { PREV = { trade_goods = precursor_relics } }
				}
				set_country_flag = gene_pillars_loot_precursor_relics
			}
			10 = { # Get Damestear
				trigger = {
					NOT = {
						has_country_flag = gene_pillars_loot_damestear
						has_country_flag = gene_pillars_loot_precursor_relics
					}
					NOT = { PREV = { trade_goods = damestear } }
				}
				set_country_flag = gene_pillars_loot_damestear
			}
			10 = {
				trigger = {
					NOT = {
						has_country_flag = gene_pillars_has_mage_A
						has_country_flag = gene_pillars_has_artificer_A
						has_country_flag = gene_pillars_loot_power_cash
					}
				}
				set_country_flag = gene_pillars_loot_power_cash
			}
			5 = {
				#TODO: Add trigger to ensure an eligible invention exists
				trigger = {
					NOT = { has_country_flag = gene_pillars_loot_insyaa_invention }
				}
				set_country_flag = gene_pillars_loot_insyaa_invention
			}
		}
	}
	[[looted]
		# subtract from the loot variable as 'earned' loot, reset the bar
		owner = { change_aPUI_var = { name = gene_pillar_loot value = -100 } }
		set_aPUI_var = { name = gene_pillar_loot which = owner }
		update_aPUI_var_bar = { name = gene_pillar_loot }
	]
}

# DO NOT USE gene_pillars_change_hp_effect DIRECTLY - add must be used for +, and lose for -
# these effects run on the country scope (the country doing the expedition), and require the event target gene_pillar_exploration_province to be saved
# NOTE: these WILL NOT fully work inside a scripted effect else_if/else/trigger_switch, as they need to fire events (for abandoning the expedition at 0 health)
gene_pillars_add_hp_effect = {
	gene_pillars_change_hp_effect = { amount = $amount$ [[overheal] overheal = yes ] add = yes }
}
gene_pillars_lose_hp_effect = {
	gene_pillars_change_hp_effect = { amount = -$amount$ lose = yes }

	# If Party is dead, abandon Exploration and don't queue another event
	if = {
		limit = {
			event_target:gene_pillar_exploration_province = {
				NOT = { aPUI_check_variable = { name = gene_pillar_health value = 1 } }
			}
			is_exploring_any_gene_pillar = yes
			# in case you trigger this twice by e.g. having two health loss effects in one option
		}
		reduce_stability_or_adm_power = yes
		set_country_flag = gene_pillars_got_killed
		gene_pillar_abandon_expedition_effect = { province_notification_event = yes }
	}
}

gene_pillars_change_hp_effect = {
	# "Change Health by X"
	tooltip = {
		change_aPUI_var = { name = gene_pillar_health value = $amount$ }
	}
	[[overheal] custom_tooltip = gene_pillars_overheal_tt ]
	[[lose]
		if = {
			limit = { NOT = { gene_pillars_has_any_frontliner_class_country = yes } }
			custom_tooltip = gene_pillars_lose_health_effect_frontliner_tt
		}
	]

	hidden_effect = {
		event_target:gene_pillar_exploration_province = {
			change_aPUI_var = { name = gene_pillar_health value = $amount$ }
			[[overheal]
				# Is Max HP Less than current HP? Set to max for an 'overheal'
				if = {
					limit = { aPUI_check_aPUI_variable = { name = gene_pillar_health which = gene_pillar_health_max } }
					set_aPUI_to_aPUI_var = { name = gene_pillar_health_max which = gene_pillar_health }
				}
			]
			[[lose]
				if = {
					limit = { # Take more damage if you have no frontliner
						#In this generic effect, check there was a hp reduction before applying again
						NOT = { owner = { gene_pillars_has_any_frontliner_class_country = yes } }
					}
					change_aPUI_var = { name = gene_pillar_health value = $amount$ }
				}
			]
			[[add]
				#Cap the health bar
				if = {
					limit = { aPUI_check_aPUI_variable = { name = gene_pillar_health which = gene_pillar_health_max } }
					set_aPUI_to_aPUI_var = { name = gene_pillar_health which = gene_pillar_health_max }
				}
			]
			update_aPUI_var_bar = { name = gene_pillar_health }
			update_aPUI_var_bar = { name = gene_pillar_health_max }
		}
	}
}

#Sets up the progress bar
gene_pillars_set_up_progress_bar = {
	# Define the increments based on the number of events
	event_target:gene_pillar_exploration_province = {
		set_aPUI_cap = { name = gene_pillar_progress amount = $events$ }
		set_aPUI_var = { name = gene_pillar_progress value = 0 }
		hidden_effect = { update_aPUI_var_bar = { name = gene_pillar_progress } }
	}
}

gene_pillar_clear_unlocked_party_setup_slot_for_injury = {
	if = {
		limit = {
			check_variable = {
				which = aPUI_gene_pillar_party_slot_framevar_$slot$
				value = @min_class_value
			}
		}
		set_variable = {
			which = aPUI_gene_pillar_party_slot_framevar_$slot$
			value = @empty
		}
	}
}

#Roll for injury on a specific class
gene_pillars_roll_injury_class_effect = {
	event_target:gene_pillar_exploration_province = {
		if = {
			limit = {
				NOT = {
					AND = {
						aPUI_check_aPUI_variable = { name = gene_pillar_health which = gene_pillar_health_max }
						aPUI_check_variable = { name = gene_pillar_health value = 1 }
					}
				} # not at max health, and at least 1 health
				# because there is an edge case here, if both vars are 0 from e.g. the province being uncolonised
			}
			if = {
				limit = { aPUI_check_variable = { name = gene_pillar_health_max value = 1 } }
				set_variable = {
					which = temp
					which = aPUI_gene_pillar_health_var
				} # the other injury effects after this need to use the same health var
				divide_variable = {
					which = temp
					which = aPUI_gene_pillar_health_max_var
				}
			}
			else = {
				set_variable = {
					which = temp
					value = 0
				}
			}
			random_list = {
				32 = {
					modifier = {
						factor = 0.9
						owner = {
							check_variable = {
								which = aPUI_gene_pillars_party_level_var
								value = 1
							}
							gene_pillars_has_any_support_class_country = yes
						}
					}
					modifier = {
						factor = 0.9
						owner = {
							check_variable = {
								which = aPUI_gene_pillars_party_level_var
								value = 2
							}
							gene_pillars_has_any_support_class_country = yes
						}
					}
					modifier = {
						factor = 0.9
						owner = {
							check_variable = {
								which = aPUI_gene_pillars_party_level_var

							gene_pillars_has_any_support_class_country = yes

								value = 4

							gene_pillars_has_any_support_class_country = yes

								value = 5
							}
							gene_pillars_has_any_support_class_country = yes

						owner = { has_country_flag = gene_pillars_has_cleric_B }

					owner = { gene_pillars_injure_class_effect = { class = $class$ } }

					}

					}

gene_pillars_generic_loot_or_trap_repeat = {
	random_list = {
		$success_factor$ = {
			gene_pillars_add_loot_effect = { amount = $party_loot$ }
			gene_pillars_add_xp_effect = { amount = $party_xp$ }
			hidden_effect = {
				change_variable = { # So I can track what to remove if the trap is sprung
					which = gene_pillars_generic_traps_and_treasure_loot_gained
					value = $party_loot$
				}
				change_variable = { # So I can track what to remove if the trap is sprung
					which = gene_pillars_generic_traps_and_treasure_xp_gained
					value = $party_xp$
				}
				change_variable = {
					which = gene_pillars_generic_traps_and_treasure_count
					value = 1
				}
			}
			if = {
				limit = {
					NOT = {
						check_variable = {
							which = gene_pillars_generic_traps_and_treasure_count
							value = 5
						}
					}
				}
				country_event = {
					id = gene_pillars.1005 # get this event again
					days = 1 # can't trigger itself without a delay
				}
			}
			# If you succeed 5 times, reset the variables in case you get this again
			else = {
				hidden_effect = {
					set_variable = {
						which = gene_pillars_generic_traps_and_treasure_loot_gained
						value = 0
					}
					set_variable = {
						which = gene_pillars_generic_traps_and_treasure_xp_gained
						value = 0
					}
					set_variable = {
						which = gene_pillars_generic_traps_and_treasure_count
						value = 0
					}
				}
			}
		}
		$fail_factor$ = {
			country_event = {
				id = gene_pillars.1006 # Trap sprung
			}
			# Reset the counter for next time we encounter this event
			hidden_effect = {
				set_variable = {
					which = gene_pillars_generic_traps_and_treasure_count
					value = 0
				}
			}
		}
	}
}
