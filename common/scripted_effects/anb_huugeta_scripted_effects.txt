S45_upgrade_mercenary_excellence = {
    hidden_effect = {
        remove_country_modifier = S45_mercenary_excellence_0
        remove_country_modifier = S45_mercenary_excellence_1
        remove_country_modifier = S45_mercenary_excellence_2
        remove_country_modifier = S45_mercenary_excellence_3
        remove_country_modifier = S45_mercenary_excellence_4
        remove_country_modifier = S45_mercenary_excellence_5
    }
    if = {
        limit = {
            check_variable = {
                which = nextRewardThreshold
                value = 101
            }
        }
        add_country_modifier = {
            name = S45_mercenary_excellence_5
            duration = 3650
        }
    }
    else_if = {
        limit = {
            check_variable = {
                which = nextRewardThreshold
                value = 100
            }
        }
        if = {
            limit = {
                check_variable = {
                    which = mercenaryMomentum
                    value = 100
                }
            }
            add_country_modifier = {
                name = S45_mercenary_excellence_5
                duration = 3650
            }
            set_variable = {
                which = baseMercenaryMomentum
                value = 25
            }
            hidden_effect = {
                set_variable = {
                    which = nextRewardThreshold
                    value = 101 # Symbolic, no more upgrades
                }
            }
        }
        else = {
            add_country_modifier = {
                name = S45_mercenary_excellence_4
                duration = 3650
            }
        }
    }
    else_if = {
        limit = {
            check_variable = {
                which = nextRewardThreshold
                value = 90
            }
        }
        if = {
            limit = {
                check_variable = {
                    which = mercenaryMomentum
                    value = 90
                }
            }
            add_country_modifier = {
                name = S45_mercenary_excellence_4
                duration = 3650
            }
            set_variable = {
                which = baseMercenaryMomentum
                value = 20
            }
            hidden_effect = {
                set_variable = {
                    which = nextRewardThreshold
                    value = 100
                }
            }
        }
        else = {
            add_country_modifier = {
                name = S45_mercenary_excellence_3
                duration = 3650
            }
        }
    }
    else_if = {
        limit = {
            check_variable = {
                which = nextRewardThreshold
                value = 80
            }
        }
        if = {
            limit = {
                check_variable = {
                    which = mercenaryMomentum
                    value = 80
                }
            }
            add_country_modifier = {
                name = S45_mercenary_excellence_3
                duration = 3650
            }
            set_variable = {
                which = baseMercenaryMomentum
                value = 15
            }
            hidden_effect = {
                set_variable = {
                    which = nextRewardThreshold
                    value = 90
                }
            }
        }
        else = {
            add_country_modifier = {
                name = S45_mercenary_excellence_2
                duration = 3650
            }
        }
    }
    else_if = {
        limit = {
            check_variable = {
                which = nextRewardThreshold
                value = 70
            }
        }
        if = {
            limit = {
                check_variable = {
                    which = mercenaryMomentum
                    value = 70
                }
            }
            add_country_modifier = {
                name = S45_mercenary_excellence_2
                duration = 3650
            }
            set_variable = {
                which = baseMercenaryMomentum
                value = 10
            }
            hidden_effect = {
                set_variable = {
                    which = nextRewardThreshold
                    value = 80
                }
            }
        }
        else = {
            add_country_modifier = {
                name = S45_mercenary_excellence_1
                duration = 3650
            }
        }
    }
    else = {
        if = {
            limit = {
                check_variable = {
                    which = mercenaryMomentum
                    value = 60
                }
            }
            add_country_modifier = {
                name = S45_mercenary_excellence_1
                duration = 3650
            }
            set_variable = {
                which = baseMercenaryMomentum
                value = 5
            }
            hidden_effect = {
                set_variable = {
                    which = nextRewardThreshold
                    value = 70
                }
            }
        }
        else = {
            add_country_modifier = {
                name = S45_mercenary_excellence_0
                duration = 3650
            }
        }
    }
}

S45_subtract_momentum = {
    custom_tooltip = S45_current_momentum_tt
    subtract_variable = {
        which = mercenaryMomentum
        value = $value$
    }
    hidden_effect = {
        # Increase score tracking for this action type
        subtract_variable = {
            which = mercActionValue_$action$
            value = $value$
        }

        if = {
            limit = {
                NOT = {
                    check_variable = {
                        which = mercenaryMomentum
                        value = 0
                    }
                }
            }
            set_variable = {
                which = mercenaryMomentum
                value = 0
            }
        }
    }
}

S45_add_momentum = {
    custom_tooltip = S45_current_momentum_tt
    change_variable = {
        which = mercenaryMomentum
        value = $value$
    }
    hidden_effect = {
        # Increase score tracking for this action type
        change_variable = {
            which = mercActionValue_$action$
            value = $value$
        }

        if = {
            limit = {
                check_variable = {
                    which = mercenaryMomentum
                    value = 101
                }
            }
            set_variable = {
                which = mercenaryMomentum
                value = 100
            }
        }
    }
}

        which = mercActionValue_condottieri_battle
        value = 0
    }

        which = mercActionValue_condottieri_siege
        value = 0

        which = mercActionValue_mercenary_victory

    set_variable = {
        which = mercActionValue_mercenary_battle

    }
    set_variable = {

    }

        value = 0
    }
    set_variable = {
        which = mercActionValue_advisor
        value = 0
    }
    set_variable = {
        which = mercActionValue_event
        value = 0
    }
    set_variable = {
        which = mercActionValue_tech
        value = 0
    }
    set_variable = {

    set_variable = {
        which = mercenaryMomentum
        which = baseMercenaryMomentum
    }
}

S45_victory_effect = {
	if = {
		limit = { 
			has_global_flag = S45_mercenary_enterprise_expensive_mechanics 
			NOT = { has_country_flag = S45_mercenary_enterprise }
			5598 = {
				owner = {
					has_country_flag = S45_mercenary_enterprise
				}
			}
		}

		5598 = {
			owner = { 
				save_event_target_as = huugeta_target
			}
		}

		# Check for Huugetan Condottieri
		if = {
			limit = {
				is_hiring_condottiere_from = event_target:huugeta_target
			}

			event_target:huugeta_target = {
				S45_add_momentum = { value = 10 action = condottieri_victory }
			}

			if = {
				limit = {
					OR = {
						ROOT = {
							any_owned_province = {
								region = telekenid_region
							}
						}
						FROM = {
							any_owned_province = {
								region = telekenid_region
							}
						}
					}
				}

				event_target:huugeta_target = {
					change_variable = {
						which = telekenidWars
						value = 1
					}
				}
			}
		}

		# Check for Huugetan Mercenaries
		if = {
			limit = {
				any_hired_mercenary_company = {
					home_province = {
						owned_by = event_target:huugeta_target
					}
				}
			}

			event_target:huugeta_target = {
				S45_add_momentum = { value = 10 action = mercenary_victory }
			}

			if = {
				limit = {
					OR = {
						ROOT = {
							any_owned_province = {
								region = telekenid_region
							}
						}
						FROM = {
							any_owned_province = {
								region = telekenid_region
							}
						}
					}
				}

				event_target:huugeta_target = {
					change_variable = {
						which = telekenidWars
						value = 1
					}
				}
			}
		}
	}
}

S45_battle_effect = {
    if = {
		limit = { 
			has_global_flag = S45_mercenary_enterprise_expensive_mechanics
			5598 = {
				owner = {
					has_country_flag = S45_mercenary_enterprise
				}
			}
		}

		5598 = {
			owner = { 
				save_event_target_as = huugeta_target
			}
		}

		if = {
			limit = {
				FROM = {
					any_war_enemy_country = {
						is_hiring_condottiere_from = event_target:huugeta_target
					}
				}
				ROOT = {
					num_of_units_in_province = {
						who = event_target:huugeta_target
						amount = 1
					}
				}
			}
			event_target:huugeta_target = {
				if = {
					limit = { NOT = { check_variable = { which = mercActionValue_condottieri_battle value = 20} } }
					S45_add_momentum = { value = 1 action = condottieri_battle }
				}
			}
		}

		if = {
			limit = {
				FROM = {
					any_war_enemy_country = {
						# Check if any enemy is hiring mercs from Huugetan lands, excluding Huugeta itself
                        NOT = { has_country_flag = S45_mercenary_enterprise }
						any_hired_mercenary_company = {
							home_province = {
								owned_by = event_target:huugeta_target
							}
							# Check if they were present at this battle
							location = { province_id = ROOT }
						}
					}
				}
			}
			event_target:huugeta_target = {
				if = {
					limit = { NOT = { check_variable = { which = mercActionValue_mercenary_battle value = 20} } }
					S45_add_momentum = { value = 1 action = mercenary_battle }
				}
			}
		}
	}
}
