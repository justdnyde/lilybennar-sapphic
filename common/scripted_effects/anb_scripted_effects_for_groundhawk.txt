# B42 - GroundHawk
B42_upgrade_marcher_goblin_level = {

	if = {
		limit = { has_country_modifier = B42_marcher_goblins_1 }
		hidden_effect = {
			remove_country_modifier = B42_marcher_goblins_1
		}
		add_country_modifier = {
			name = B42_marcher_goblins_2
			duration = -1
		}
		
	}
	else_if = {
		limit = { has_country_modifier = B42_marcher_goblins_2 }
		hidden_effect = {
			remove_country_modifier = B42_marcher_goblins_2
		}
		
		add_country_modifier = {
			name = B42_marcher_goblins_3
			duration = -1
		}
	}
	else_if = {
		limit = { has_country_modifier = B42_marcher_goblins_3 }
		hidden_effect = { remove_country_modifier = B42_marcher_goblins_3 }
		add_country_modifier = {
			name = B42_marcher_goblins_4
			duration = -1
		}
	}
	else_if = {
		limit = { has_country_modifier = B42_marcher_goblins_4 }
		hidden_effect = { remove_country_modifier = B42_marcher_goblins_4 }
		add_country_modifier = {
			name = B42_marcher_goblins_5
			duration = -1
		}
	}
	else_if = {
		limit = { has_country_modifier = B42_marcher_goblins_5 }
		hidden_effect = { remove_country_modifier = B42_marcher_goblins_5 }
		add_country_modifier = {
			name = B42_marcher_goblins_6
			duration = -1
		}
	}
}

B42_rewards_for_last_row_missions = {
	if = {
		limit = {
			check_variable = {
				which = last_row_counter
				value = 4
			}
		}
		add_war_exhaustion = -5
		add_yearly_manpower = 0.833
	}
	else_if = {
		limit = {
			check_variable = {
				which = last_row_counter
				value = 3
			}
		}	
		add_war_exhaustion = -4
		add_yearly_manpower = 0.667
	}
	else_if = {
		limit = {
			check_variable = {
				which = last_row_counter
				value = 2
			}
		}	
		add_war_exhaustion = -3
		add_yearly_manpower = 0.5		
	}
	else_if = {
		limit = {
			check_variable = {
				which = last_row_counter
				value = 1
			}
		}	
		add_war_exhaustion = -2
		add_yearly_manpower = 0.333		
	}
	else = {
		add_war_exhaustion = -1
		add_yearly_manpower = 0.167
	}
	hidden_effect = {
		change_variable = {
			which = last_row_counter
			value = 1
		}	
	}

}
B42_refresh_cb_for_missions = {
	custom_tooltip = B42_refresh_cb_for_missions_tt
	hidden_effect = {
		if = {
			limit = {
				NOT = { mission_completed = B42_infernalist_cults }
			}
			escann_superregion = {
				limit = {
					OR = {
						religion = infernal_court
						religion = xhazobkult
					}
				}
				add_claim = ROOT
			}
			every_country = {
				limit = {
					capital_scope = { continent = europe }
					OR = {
						religion = infernal_court
						religion = xhazobkult
					}
				}
				set_country_flag = B42_religous_target
				reverse_add_casus_belli = {
					target = ROOT
					type = cb_crusade_mission
					months = 3600
				}
			}
		}
		if = {
			limit = {
				NOT = { mission_completed = B42_safeguard_the_caves }
			}
			every_country = {
				limit = {
					capital_scope = { superregion = west_dwarovar_superregion }
					OR = {
						has_country_modifier = racial_pop_goblin_expulsion
						has_country_modifier = racial_pop_goblin_purge
					}
				}
				
				reverse_add_casus_belli = {
					target = ROOT
					type = cb_liberate_mission
					months = 3600
				}
			}
			every_owned_province = {
				every_neighbor_province = {
					limit = {
						NOT = { owned_by = ROOT }
						superregion = west_dwarovar_superregion
						NOT = { fast_has_pop_of_race = { race = goblin } }
						NOT = { owner = { ruler_is_goblin = yes } }
					}
					add_claim = ROOT
				}
			}
		}
		if = {
			limit = {
				NOT = { mission_completed = B42_free_the_woods }
			}
			every_country = {
				limit = {
					capital_scope = { superregion = deepwoods_superregion }
					religion_group = fey_religion
				}
				set_country_flag = B42_religous_target
				reverse_add_casus_belli = {
					target = ROOT
					type = cb_crusade_mission
					months = 3600
				}
			}
		}
	}
}
