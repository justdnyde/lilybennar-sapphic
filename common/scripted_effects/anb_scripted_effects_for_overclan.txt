overclan_clear_discontent_bureaucrats_effect = {
	remove_province_modifier = F78_discontent_bureaucrats_1
	remove_province_modifier = F78_discontent_bureaucrats_2
	remove_province_modifier = F78_discontent_bureaucrats_3
	remove_province_modifier = F78_discontent_bureaucrats_4
	remove_province_modifier = F78_discontent_bureaucrats_5
}

overclan_increase_discontent_bureaucrats_effect = {
	if = {
		limit = { NOT = { has_province_modifier = F78_ca_overboss } }
		#overboss means they're content
		if = {
			limit = { has_province_modifier = F78_discontent_bureaucrats_1 }
			remove_province_modifier = F78_discontent_bureaucrats_1
			add_province_modifier = {
				name = F78_discontent_bureaucrats_2
				duration = -1
			}
		}
		else_if = {
			limit = { has_province_modifier = F78_discontent_bureaucrats_2 }
			remove_province_modifier = F78_discontent_bureaucrats_2
			add_province_modifier = {
				name = F78_discontent_bureaucrats_3
				duration = -1
			}
		}
		else_if = {
			limit = { has_province_modifier = F78_discontent_bureaucrats_3 }
			remove_province_modifier = F78_discontent_bureaucrats_3
			add_province_modifier = {
				name = F78_discontent_bureaucrats_4
				duration = -1
			}
		}
		else_if = {
			limit = { has_province_modifier = F78_discontent_bureaucrats_4 }
			remove_province_modifier = F78_discontent_bureaucrats_4
			add_province_modifier = {
				name = F78_discontent_bureaucrats_5
				duration = -1
			}
		}
		else_if = {
			limit = { NOT = { has_province_modifier = F78_discontent_bureaucrats_5 } }
			#not max, and not 1-4 means no modifier
			add_province_modifier = {
				name = F78_discontent_bureaucrats_1
				duration = -1
			}
		}
	}
}

spend_adm_development = {
	for = {
		amount = $amount$
		effect = "
			random_list = {
				1000 = {
					trigger = { base_tax = 2 }
					add_base_tax = -1
				}
				1 = {
					trigger = { base_production = 2 NOT = { base_tax = 2 } }
					add_base_production = -1
				}
				1 = {
					trigger = { base_manpower = 2 NOT = { base_tax = 2 } }
					add_base_manpower = -1
				}
			}
		"
	}
}

overclan_create_voltoj_academy = { #province
	if = {
		limit = { province_group = hold_province }
		add_permanent_province_modifier = {
			name = F78_small_voltoj_academy_hold
			duration = -1
		}
	}
	else = {
		add_permanent_province_modifier = {
			name = F78_small_voltoj_academy
			duration = -1
		}
	}

	if = {
		limit = { owner = { has_estate_privilege = estate_voltoj_academy_privileges } }
		add_province_modifier = {
			name = f78_academy_privs
			duration = -1
		}
	}
	if = {
		limit = { owner = { has_enacted_parliament_action = { parliament_action = overclan_improve_city_accessibility_act } } }
		add_harpy_minority_size_effect = yes
		add_province_modifier = {
			name = F78_harpy_upper_cities_mod
			duration = -1
		}
	}
}
