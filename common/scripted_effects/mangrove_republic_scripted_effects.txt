mangrove_republic_setup = {
	# check for end of reclamation
	if = {
		limit = { NOT = { has_global_flag = mangrove_reclamation_done } }
		add_country_modifier = {
			name = mangrove_peace
			duration = -1
		}
		hidden_effect = {
			mangrove_gain_cb = yes
			set_variable = { mangrove_punia_logi = 0 }
		}
	}

	if = {
		limit = { ai = no }
		every_owned_province = {
			limit = {
				region = tagaala_mangroves_region
			}
			add_province_modifier = {
				name = mangrove_bio_infra_decay
				duration = -1
			}
		}
	}

	custom_tooltip = mangrove_capstone_province_tt
	
	hidden_effect = {
		every_known_country = {
			limit = { has_government_attribute = mangrove_republic }
			add_opinion = {
				who = ROOT
				modifier = mangrove_republic_relations
			}
			reverse_add_opinion = {
				who = ROOT
				modifier = mangrove_republic_relations
			}

		}
		set_variable = { mangrove_bio_infra_finished_count = 0 }
	}
}

mangrove_on_province_owner_change_effect = {
	if = {
		limit = { region = tagaala_mangroves_region }
		clr_province_flag = mangrove_bio_infra_finished
		clr_province_flag = mangrove_bio_infra_stage_4
		clr_province_flag = mangrove_bio_infra_stage_3
		clr_province_flag = mangrove_bio_infra_stage_2
		clr_province_flag = mangrove_bio_infra_stage_1
		clr_province_flag = mangrove_bio_infra_path_chosen
		if = { # AI doesn't get dev cost malus
			limit = {
				owner = {
					ai = no
					has_government_attribute = mangrove_republic
				}
			}
			add_province_modifier = {
				name = mangrove_bio_infra_decay
				duration = -1
			}
		}
		if = {
			limit = { has_saved_global_event_target = mangrove_target_province }
			if = { # non mangrove republic has conquered/colonized a mangrove province
				limit = {
					province_id = event_target:mangrove_target_province
					owner = { NOT = { has_government_attribute = mangrove_republic } }
				}
				remove_province_modifier = mangrove_reclamation_site
				clear_global_event_target = mangrove_target_province
				clr_global_flag = mangrove_reclaiming
				set_global_flag = mangrove_restart_voting
				mangrove_event = { COUNTRY_EVENT = "mangrove.122" }
			}
		}
	}
}

# places value in mangrove_true_influence
mangrove_calc_true_influence = {
	export_to_variable = {
		which = mangrove_true_influence
		value = modifier:diplomatic_reputation
	}
	multiply_variable = { mangrove_true_influence = 4 }
	# true mangrove influence = mangrove influence + 4 * diplo rep
	change_variable = {
		which = mangrove_true_influence
		which = mangrove_influence
	}
}

mangrove_new_leader_on_annexed = {
	random_known_country = {
		limit = { has_government_attribute = mangrove_republic }
		save_global_event_target_as = mangrove_leader
	}
	every_known_country = {
		limit = { has_government_attribute = mangrove_republic }
		if = {
			limit = {
				event_target:mangrove_leader = {
					NOT = {
						check_variable = {
							which = mangrove_influence
							which = PREV
						}
					}
				}
			}
			save_global_event_target_as = mangrove_leader
		}
	}
	mangrove_change_ownership = yes
	mangrove_event = { COUNTRY_EVENT = "mangrove.123" }
}

mangrove_change_ownership = {
	if = {
		limit = {
			has_saved_global_event_target = mangrove_target_province
			event_target:mangrove_target_province = {
				owner = {
					NOT = { tag = event_target:mangrove_leader } # don't change anything if leader hasn't changed
				}
			}
		}
		hidden_effect = {
			event_target:mangrove_leader = {
				event_target:mangrove_target_province = {
					remove_core = owner
					add_core = PREV
					cede_province = PREV
				}
			}
		}
	}
}

mangrove_gain_influence = {
	change_variable = { mangrove_influence = $INFLUENCE$ }
	if = {
		limit = { check_variable = { mangrove_influence = 100 } }
		set_variable = { mangrove_influence = 100 }
	}
	if = {
		limit = { has_saved_global_event_target = mangrove_leader }
		if = {
			limit = {
				NOT = { tag = event_target:mangrove_leader }
				event_target:mangrove_leader = {
					NOT = {
						check_variable = {
							which = mangrove_influence
							which = PREV
						}
					}
				}
			}
			save_global_event_target_as = mangrove_leader
			mangrove_change_ownership = yes
			mangrove_event = { COUNTRY_EVENT = "mangrove.123" }
		}
	}
	else = {
		save_global_event_target_as = mangrove_leader
	}
}

mangrove_lose_influence = {
	subtract_variable = { mangrove_influence = $INFLUENCE$ }
	if = {
		limit = { NOT = { check_variable = { mangrove_influence = 0 } } }
		set_variable = { mangrove_influence = 0 }
	}
	if = {
		limit = { has_saved_global_event_target = mangrove_leader }
		if = {
			limit = { tag = event_target:mangrove_leader }
			every_known_country = {
				limit = { has_government_attribute = mangrove_republic }
				if = {
					limit = {
						event_target:mangrove_leader = {
							NOT = {
								check_variable = {
									which = mangrove_influence
									which = PREV
								}
							}
						}
					}
					save_global_event_target_as = mangrove_leader
				}
			}
			if = {
				limit = { NOT = { tag = event_target:mangrove_leader } }
				mangrove_change_ownership = yes
				mangrove_event = { COUNTRY_EVENT = "mangrove.123" }
			}
		}
	}
}

	}

					which = mangrove_donate_every_3

				}
			}

		}
	}
}

mangrove_advance_bio_infra_stage = {
	# after choosing 1st upgrade
	if = {
		limit = { has_province_flag = mangrove_bio_infra_stage_1 }
		clr_province_flag = mangrove_bio_infra_stage_1
		set_province_flag = mangrove_bio_infra_stage_2
		add_province_modifier = { # fake modifier to show all upgrades
			name = mangrove_bio_infra_dummy_mod
			duration = -1
		}

	# after choosing 2nd upgrade
	else_if = {
		limit = { has_province_flag = mangrove_bio_infra_stage_2 }
		clr_province_flag = mangrove_bio_infra_stage_2
		set_province_flag = mangrove_bio_infra_stage_3

	# after choosing 3rd upgrade

		limit = { has_province_flag = mangrove_bio_infra_stage_3 }

		set_province_flag = mangrove_bio_infra_stage_4
		set_province_flag = mangrove_bio_infra_$PATH$_minor
	}
}

			limit = { FROM = { has_country_flag = mangrove_capstone_unlocked } }

		if = {
			limit = { is_capital = yes }

		}
	}
	else = {
		tooltip = {
			mangrove_bio_infra_add_auxiliary_effect = yes
		}
	}
}
