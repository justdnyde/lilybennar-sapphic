harpylen_new_heir = {
	if = {
		limit = { mission_completed = F98_godking_dragon }
		random_list = {
			1 = { add_heir_personality = iron_crowned_personality }
			1 = { add_heir_personality = peasant_revolt_leader_personality }
			1 = { add_heir_personality = great_engineer_personality }
			1 = { add_heir_personality = legendary_conqueror_personality }
		}
	}
	if = {
		limit = { mission_completed = F98_true_queendom }
		if = {
			limit = { religion = the_hunt }
			add_country_modifier = {
				name = harpy_crane_doctrine_heir
				duration = -1
			}
			add_country_modifier = {
				name = harpy_myna_doctrine_heir
				duration = -1
			}
			add_country_modifier = {
				name = harpy_eagle_doctrine_heir
				duration = -1
			}
		}
		else = {
			hidden_effect = {
				while = {
					limit = { NOT = { heir_adm = 5 } }
					change_heir_adm = 1
				}
				while = {
					limit = { NOT = { heir_dip = 5 } }
					change_heir_dip = 1
				}
				while = {
					limit = { NOT = { heir_mil = 5 } }
					change_heir_mil = 1
				}
			}
		}
	}
	if = {
		limit = { mission_completed = F98_pride_of_dynasty }
		random = {
			chance = 40
			define_powerful_mage = { type = heir }
		}
		random_list = {
			1 = {
				trigger = { NOT = { heir_adm = 6 } }
				change_heir_adm = 1
			} 
			1 = {
				trigger = { NOT = { heir_dip = 6 } }
				change_heir_dip = 1
			}
			1 = {
				trigger = { NOT = { heir_mil = 6 } }
				change_heir_mil = 1
			}
		}
	}
}

harpylen_formation_effect = {
	override_country_name = HARPYLEN_NAME
	hidden_effect = {
		every_owned_province = {
			limit = { is_part_of_hre = yes }
			set_in_empire = no
		}
	}
	if = {
		limit = { is_or_was_tag = { tag = F50 } }
		hidden_effect = {
			add_harpy_queendom_reform = yes
			add_government_reform = merchant_matriarchs_harpylen_reform
		}
		custom_tooltip = mulen_harpylen_tt
		add_country_modifier = {
			name = mulen_joint_rule
			duration = -1
		}		
	}
	else = {
		change_tag = F98
		swap_non_generic_missions = yes
		hidden_effect = {
			set_variable = {
				which = F98_harpy_spells_cast
				value = 0
			}
			set_variable = {
				which = F98_hunt_power_used_count
				value = 0
			}
			set_variable = {
				which = F98_render_unto_total_dev_transferred
				value = 0
			}
		}
		set_country_flag = count_battles_won_with_ruler_leading
	}
	country_event = { id = harpy_events.123 }
	set_government_rank_scripted_effect = { rank = 3 }
	add_prestige = 50
	if = {
		limit = { has_custom_ideas = no }
		country_event = { id = ideagroups.1 } #Swap Ideas
	}
	add_country_modifier = {
		name = "centralization_modifier"
		duration = 7300
	}
	lock_racial_admin = yes
}
