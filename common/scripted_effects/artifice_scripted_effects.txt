update_research_speed = {
	hidden_effect = {
		if = {
			limit = {
				ai = yes
			}
			set_variable = {
				which = ResearchSpeed
				value = 10
			}
		}
		else_if = {
			limit = {
				has_country_flag =  artifice_research_random
			}
			set_variable = {
				which = ResearchSpeed
				value = 5
			}
		}
		else_if = {
			limit = {
				has_country_flag = artifice_research_slow
			}
			set_variable = {
				which = ResearchSpeed
				value = 15
			}
		}
		else_if = {
			limit = {
				has_country_flag = artifice_research_fast
			}
			set_variable = {
				which = ResearchSpeed
				value = 5
			}
		}
		else = {
			set_variable = {
				which = ResearchSpeed
				value = 10
			}
		}

		if = { #increase base research time
			limit = {
				has_country_flag = increased_artifice_research_5
			}
			change_variable = {
				which = ResearchSpeed
				value = 5
			}
		}

		# new_years = base_years * (1-x)
		# example: -20% research time -> 10 years * (1-0.20) = 8 years
		export_to_variable = {
			which = ResearchSpeedMultiplier
			value = modifier:artificers_research_time
		}
		change_variable = {
			which = ResearchSpeedMultiplier
			value = 1
		}
		if = { # clamp to 0.20 to avoid negative research time
			limit = {
				NOT = {
					check_variable = {
						which = ResearchSpeedMultiplier
						value = 0.20
					}
				}
			}
			set_variable = {
				which = ResearchSpeedMultiplier
				value = 0.20
			}
		}
		multiply_variable = {
			which = ResearchSpeed
			which = ResearchSpeedMultiplier
		}
	}
}

set_up_artifice_system = {
	if = {
		limit = {
			NOT = {
				has_country_flag = artifice_set_up
			}
		}
		set_variable = {
			which = StaticArtificePoints
			value = 0
		}
		set_variable = {
			which = ArtificeUpkeep
			value = 0
		}
		#sets initial tier
		set_country_flag = artifice_unlocked_basic

		#tracks when you unlock a new tier
		set_variable = {
			which = artifice_level_progress
			value = 0
		}

		#tracks researches done before 1650
		set_variable = {
			which = ResearchArtificePoints
			value = 0
		}

		#sets up the maximum allowed simultainious active inventions
		set_variable = {
			which = artificers_invention_slots_base
			value = 7
		}
		set_country_flag = artifice_set_up
	}
	if = {
		limit = {
			NOT = { has_country_flag = artificery_unlocked }
			NOT = { has_country_flag = artificery_choice }
			NOT = { tag = H74 } #dak special
			OR = {
				is_artificery_culture_government = yes
				AND = {
					has_institution = global_trade #Submod changed from manufactories to 50 years sooner
					OR = {
						NOT = { religion = bulwari_sun_cult }
						is_chosen_country = no
					}
				}
			}
		}
		country_event = {
			id = magic_artificery_setup.1
		}	
		set_country_flag = artificery_choice
	}
	else_if = {
		limit = {
			OR = {
				has_estate_privilege = estate_artificers_organization_pillar_expedition
				has_estate_privilege = estate_artificers_organization_mechanim_base
			}
			has_institution = manufactories
			NOT = { has_country_flag = artificery_choice }
		}
		remove_estate_privilege = estate_artificers_organization_pillar_expedition
		remove_estate_privilege = estate_artificers_organization_mechanim_base

		set_country_flag = artificery_choice
		country_event = { id = magic_artificery_setup.2 }
	}
	else_if = {
		limit = { #you have no flags and are a spawnable
			is_not_in_mage_mode = yes
			is_not_in_mixed_mode = yes
			is_not_in_artificery_mode = yes
		}
		set_country_flag = magic_artificery_mage_only
	}
	update_artifice_points = yes
}

create_leader_scaling_with_tradition_artifice = {
	if = {
		limit = {
			$type$_tradition = 80
		}
		create_$leader$ = {
			tradition = 100
			add_fire = 3
			add_siege = 1
		}
	}
	else_if = {
		limit = {
			$type$_tradition = 70
		}
		create_$leader$ = {
			tradition = 90
			add_fire = 3
			add_siege = 1
		}
	}
	else_if = {
		limit = {
			$type$_tradition = 60
		}
		create_$leader$ = {
			tradition = 80
			add_fire = 3
			add_siege = 1
		}
	}
	else_if = {
		limit = {
			$type$_tradition = 50
		}
		create_$leader$ = {
			tradition = 70
			add_fire = 2
			add_siege = 1
		}
	}
	else_if = {
		limit = {
			$type$_tradition = 40
		}
		create_$leader$ = {
			tradition = 60
			add_fire = 2
			add_siege = 1
		}
	}
	else_if = {
		limit = {
			$type$_tradition = 30
		}
		create_$leader$ = {
			tradition = 50
			add_fire = 2
		}
	}
	else = {
		create_$leader$ = {
			tradition = 40
			add_fire = 2
		}
	}
	custom_tooltip = create_leader_scaling_with_tradition_tt
}

	if = {
		limit = {
			has_country_flag = artifice_research_slow
		}
		if = {
			limit = {
				is_year = 1500
				has_country_flag = artifice_unlocked_basic
				NOT = { has_country_flag = artifice_limited_to_tier_1 }
				check_variable = {
					which = artifice_level_progress
					value = 2
				}
			}
			set_country_flag = artifice_unlocked_advanced
			clr_country_flag = artifice_unlocked_basic
			set_variable = {
				which = artifice_level_progress
				value = 0
			}
		}
		else_if = {
			limit = {
				NOT = { is_year = 1500 }
				has_country_flag = artifice_unlocked_basic
				NOT = { has_country_flag = artifice_limited_to_tier_1 }
				check_variable = {
					which = artifice_level_progress
					value = 4
				}
			}
			set_country_flag = artifice_unlocked_advanced
			clr_country_flag = artifice_unlocked_basic
			set_variable = {
				which = artifice_level_progress
				value = 0
			}
		}
		else_if = {			
			limit = { 
				is_year = 1550
				has_country_flag = artifice_unlocked_advanced
				NOT = { has_country_flag = artifice_limited_to_tier_2 }

					which = artifice_level_progress
					value = 2
				}
			}
			set_country_flag = artifice_unlocked_expert
			clr_country_flag = artifice_unlocked_advanced
			set_variable = {
				which = artifice_level_progress
				value = 0
			}
		}
		else_if = {
			limit = { 
				NOT = { is_year = 1550 }
				has_country_flag = artifice_unlocked_advanced

	}

		limit = {
			OR = {

artifice_magic_switch_to_artificery_only_mode = {
	custom_tooltip = force_artificery_only_tt
	if = {
		limit = { NOT = { has_country_flag = magic_artificery_artificery_only } }
		hidden_effect = {
			clr_country_flag = magic_artificery_mage_only
			set_country_flag = magic_artificery_artificery_only
			clr_country_flag = magic_artificery_mixed
			remove_country_modifier = magic_artificery_mixed_mod
		
			# Ends whatever research you're doing
			country_event = { id = magic_system.99 }
			
			#bans the mages
			country_event = { 
				id = magic_artificery_setup.3
			}
			#unlocks artificery if you havent by now
			if = {
				limit = { 
					NOT = { has_country_flag = artificery_unlocked }
				}
				set_country_flag = artificery_unlocked
				country_event = {
					id = magic_artificery_setup.2
					days = 1
				}
			}
			if = { #To prevent player from picking magic only mode if they previously decided to have artificery 
				limit = { #Losing artificery after gaining it is unsupported in the system and has undefined behavior involved (for example, what happens to currently-researching inventions).
					has_country_flag = unlocked_artificery_by_pillar
				}
				set_country_flag = artificery_needed
			}
		}
	}
}

mechanim_automatory_cooldown_effect = {
	if = {
		limit = {
			OR = {
				has_country_flag = mechanim_automatory_building_working_mechanim
				has_country_flag = mechanim_automatory_building_working_privilege
			}
			automatory = 1
			NOT = {
				has_country_modifier = racial_pop_mechanim_expulsion
				has_country_modifier = racial_pop_mechanim_purge
			}
		}
		if = {
			limit = {
				OR = {
					has_estate_privilege = artifice_invention_industrial_automatories
					NOT = { has_country_flag = automatory_cooldown }
				}
			}
			set_country_flag = automatory_cooldown
			if = {
				limit = { ai = yes }
				mechanim_automatory_cooldown_count_ai = yes
			}
			else = {
				mechanim_automatory_cooldown_count_player = yes
			}
		}
		else = {
			clr_country_flag = automatory_cooldown
		}
	}
}

mechanim_automatory_spread = {
	export_to_variable = {
		which = automatory_number_of_provinces_without_mechanim
		value = trigger_value:num_of_cities 
	}
	if = {
		limit = {
			check_variable = { #Checking if variable is set 
				which = mechanim_count
				value = 1
			}
		}
		multiply_variable = { #Trying to account for small minorities
			which = automatory_number_of_provinces_without_mechanim
			value = 1.2
		}
		subtract_variable = {
			which = automatory_number_of_provinces_without_mechanim
			which = mechanim_count
		}
	}
	if = {
		limit = {
			check_variable = {
				which = mechanim_spread_country
				which = mechanim_spread_per_year
			}
		}
		subtract_variable = {
			which = mechanim_spread_country
			which = mechanim_spread_per_year
		}
		set_variable = {
			which = mechanim_spread_current
			which = mechanim_spread_per_year
		}
	}
	else = {
		set_variable = {
			which = mechanim_spread_current
			which = mechanim_spread_country
		}
		set_variable = {
			which = mechanim_spread_country
			value = 0
		}
	}
	if = { #Small amount of Mechanims in our country
		limit = {
			check_variable = {
				which = automatory_number_of_provinces_without_mechanim
				which = mechanim_count
			}
		}
		while = {
			limit =  {
				check_variable = {
					which = mechanim_spread_current
					value = 1
				}
			}
			subtract_variable = {
				which = mechanim_spread_current
				value = 1
			}
			mechanim_automatory_spread_for_player = yes
		}
	}
	else = { #There is a lot of mechanims in our country
		while = {
			limit =  {
				check_variable = {
					which = mechanim_spread_current
					value = 1
				}
			}
			subtract_variable = {
				which = mechanim_spread_current
				value = 1
			}
			reverse_mechanim_automatory_spread_for_player = yes
		}
	}
	clr_country_flag = mechanim_automatory_spread_finish_while_with_else
}
mechanim_automatory_spread_for_player = {
	if = {
		limit = {
			NOT = { has_country_flag = mechanim_automatory_spread_finish_while_with_else }
		}
		random_owned_province = {
			limit = {
				OR = {
					culture_group = mechanim
					has_large_mechanim_minority_trigger = yes
				}
				any_neighbor_province = {
					owned_by = ROOT
					NOT = {
						culture_group = mechanim
						has_large_mechanim_minority_trigger = yes
					}
				}
			}
			random_neighbor_province = {
				limit = {
					owned_by = ROOT
					NOT = {
						culture_group = mechanim
						has_large_mechanim_minority_trigger = yes
					}
				}
				save_event_target_as = mechanim_spread_province
				owner = { set_country_flag = found_mechanim_spread_province }
			}
		}
		if = { #If failed to find province to do normal spread finish with else
			limit = { NOT = { has_country_flag = found_mechanim_spread_province } } #Cant use has_saved_event_target because we could see province from previous while
			set_country_flag = mechanim_automatory_spread_finish_while_with_else #Cleared once while ends
			random_owned_province = {
				save_event_target_as = mechanim_spread_province
			}
		}
		else = {
			clr_country_flag = found_mechanim_spread_province
		}
	}
	else = {
		random_owned_province = {
			save_event_target_as = mechanim_spread_province
		}
	}
	event_target:mechanim_spread_province = {
		#Main effect
		mechanim_automatory_main_effect = yes
		#Additional effects
		mechanim_automatory_additional_effects = yes
	}
}
