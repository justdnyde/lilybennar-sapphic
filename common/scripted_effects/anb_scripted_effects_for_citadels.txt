# They start at tech 2 so I only need to start checking tech 3
# I will assume they will be done with this system before getting to admin tech 24
insyaa_charkuchin_kobold_citadel_supplies_tech = {
	# increment the variable tracking monthly gain
	change_variable = {
		which = insyaa_charkuchin_kobold_citadel_supplies_adm_tech_gain
		value = 2.5
	}
	change_variable = {
		which = insyaa_charkuchin_kobold_citadel_supplies_total_gain
		value = 2.5
	}
}

# FROM is the previous owner, ROOT is the province
insyaa_charkuchin_kobold_citadel_supplies_update_count_remove = {
	# First, check if citadel. That will then exclude everything else
	# Then, check if colony. That will exclude everything else, since you can't build SFs on colonies
	# Then, check if SF
	# Then, check if city
	if = {
		limit = { province_group = citadel_group_main NOT = { has_province_modifier = charkuchin_kobold_citadel_0 } }
		if = {
			limit = {
				OR = {
					has_province_flag = kobold_citadel_level_1
					has_province_flag = kobold_citadel_level_2
				}
			}
			FROM = { #Citadel
				change_variable = {
					which = insyaa_charkuchin_kobold_citadel_supplies_citadel_gain
					value = -5
				}
				change_variable = {
					which = insyaa_charkuchin_kobold_citadel_supplies_total_gain
					value = -5
				}
			}
		}
		else_if = {
			limit = {
				OR = {
					has_province_flag = kobold_citadel_level_3
					has_province_flag = kobold_citadel_level_4
				}
			}
			FROM = { #Citadel
				change_variable = {
					which = insyaa_charkuchin_kobold_citadel_supplies_citadel_gain
					value = -10
				}
				change_variable = {
					which = insyaa_charkuchin_kobold_citadel_supplies_total_gain
					value = -10
				}
			}
		}
		else_if = {
			limit = {
				has_province_flag = kobold_citadel_level_5
			}
			FROM = {
				change_variable = {
					which = insyaa_charkuchin_kobold_citadel_supplies_citadel_gain
					value = -15
				}
				change_variable = {
					which = insyaa_charkuchin_kobold_citadel_supplies_total_gain
					value = -15
				}
			}
		}
	}
	else_if = {
		limit = { # Colonies are cheaper and tracked separately
			OR = { # I use this because when on_province_owner_change triggers on a new colony
			# you start is_colony = yes is still false
				AND = { # This will be when a colony changes hands
					owner = { exists = yes }
					is_city = no
				}
				has_province_modifier = supplies_colony # This will be when it is decolonised
			}
		}
		remove_province_modifier = supplies_colony # Remove it in case the conquerer doensn't use supplies. If they do, it will be added back immediately
		if = { # Different regions have different costs so we track them differently
			limit = { superregion = charkuchar_superregion }
			FROM = {
				insyaa_charkuchin_kobold_citadel_supplies_update_counter = {
					region_group = charkuchar_range
					location_type = colony
					value = -1
					cost = -0.75
				}
			}
		}
		else_if = {
			limit = {
				superregion = insyaan_lowlands_superregion
				NOT = { region = tagaala_mangroves_region }
			}
			FROM = {
				insyaa_charkuchin_kobold_citadel_supplies_update_counter = {
					region_group = lowlands
					location_type = colony
					value = -1
					cost = -3
				}
			}
		}
		else_if = {
			limit = {
				superregion = hinuilands_superregion
			}
			FROM = {
				insyaa_charkuchin_kobold_citadel_supplies_update_counter = {
					region_group = titanoflora
					location_type = colony
					value = -1
					cost = -15
				}
			}
		}
	}
	else_if = { #Charkuchin Kobold supply centres
		limit = {
			has_province_modifier = charkuchin_kobold_supply_facility
		}
		FROM = {
			change_variable = {
				which = insyaa_charkuchin_kobold_citadel_supplies_supply_facility_count
				value = -1
			}
			change_variable = {
				which = insyaa_charkuchin_kobold_citadel_supplies_total_gain
				value = -10
			}
		}
	}
	# The script for adding uses slightly different logic and an "if" not "else" since you might conquer a SF but have it deleted
	# That is not a concern when you lose a province
	else = { # Otherwise, it's a province we need to spend monthly supplies on
		if = { # Different regions have different costs so we track them differently
			limit = { superregion = charkuchar_superregion }
			FROM = {
				insyaa_charkuchin_kobold_citadel_supplies_update_counter = {
					region_group = charkuchar_range
					location_type = city
					value = -1
					cost = -1.5
				}
			}
		}
		else_if = {
			limit = {
				superregion = insyaan_lowlands_superregion
				NOT = { region = tagaala_mangroves_region }
			}
			FROM = {
				insyaa_charkuchin_kobold_citadel_supplies_update_counter = {
					region_group = lowlands
					location_type = city
					value = -1
					cost = -6
				}
			}
		}
		else_if = {
			limit = {
				superregion = hinuilands_superregion
			}
			FROM = {
				insyaa_charkuchin_kobold_citadel_supplies_update_counter = {
					region_group = titanoflora
					location_type = city
					value = -1
					cost = -30
				}
			}
		}
	}
}

insyaa_charkuchin_kobold_citadel_supplies_update_count_add = {

	# Then, check if colony. That will exclude everything else, since you can't build SFs on colonies
	# Then, check if SF. It might be kept, it might not
	# Then, check if city

	if = {
		limit = { province_group = citadel_group_main NOT = { has_province_modifier = charkuchin_kobold_citadel_0 } }
		if = {
			limit = {
				OR = {
					has_province_flag = kobold_citadel_level_1
					has_province_flag = kobold_citadel_level_2
				}
			}
			owner = { #Citadel
				change_variable = {
					which = insyaa_charkuchin_kobold_citadel_supplies_citadel_gain
					value = 5
				}
				change_variable = {
					which = insyaa_charkuchin_kobold_citadel_supplies_total_gain
					value = 5
				}
			}
		}
		else_if = {
			limit = {
				OR = {
					has_province_flag = kobold_citadel_level_3
					has_province_flag = kobold_citadel_level_4
				}
			}
			owner = { #Citadel
				change_variable = {
					which = insyaa_charkuchin_kobold_citadel_supplies_citadel_gain
					value = 10
				}
				change_variable = {
					which = insyaa_charkuchin_kobold_citadel_supplies_total_gain
					value = 10
				}
			}
		}
		else_if = {
			limit = {
				has_province_flag = kobold_citadel_level_5

					which = insyaa_charkuchin_kobold_citadel_supplies_citadel_gain

					value = 15
				}
			}
		}
	}
	else_if = {
		limit = { # Colonies are cheaper and tracked separately
			# I use this because when on_province_owner_change triggers on a new colony
			# you start is_colony = yes is still false
			# This will be when a colony changes hands. Since this is for the adding, I don't
			# need to worry about the new owner being empty (from decolonisation)
			owner = { exists = yes }
			is_city = no
		}

		# First, set a mod so when the province is decolonised we know what it was before
		# Can't use a province flag since those are cleared on decolonisation
		add_permanent_province_modifier = {
			name = supplies_colony
			duration = -1
		}
		if = { # Different regions have different costs so we track them differently
			limit = { superregion = charkuchar_superregion }
			owner = {
				insyaa_charkuchin_kobold_citadel_supplies_update_counter = {
					region_group = charkuchar_range
					location_type = colony
					value = 1
					cost = 0.75
				}
			}
		}
		else_if = {
			limit = {
				superregion = insyaan_lowlands_superregion
				NOT = { region = tagaala_mangroves_region }
			}
			owner = {
				insyaa_charkuchin_kobold_citadel_supplies_update_counter = {
					region_group = lowlands
					location_type = colony
					value = 1
					cost = 3
				}
			}
		}

			limit = {
				superregion = hinuilands_superregion
			}
			owner = {
				insyaa_charkuchin_kobold_citadel_supplies_update_counter = {
					region_group = titanoflora
					location_type = colony
					value = 1
					cost = 15
				}
			}
		}
	}
	else_if = { #Charkuchin Kobold supply centres. Leave it be if the new owner doesn't use it
		limit = {
			has_province_modifier = charkuchin_kobold_supply_facility
		}
		if = {
			limit = {
				OR = {
					owner = { #You can only have 10
						check_variable = {
							which = insyaa_charkuchin_kobold_citadel_supplies_supply_facility_count
							value = 10
						}
					}
					NOT = { #The AI can build them more easily, so check it's valid for a human
						ai = yes
						AND = {
							base_production = 4
							has_fort_building_trigger = yes
							has_production_building_trigger = yes	
						}
					}
				}
			}
			remove_province_modifier = charkuchin_kobold_supply_facility
			province_event = { id = charkuchin_kobold.11 }
		}
		else = { # else increase SF count by 1
			owner = {
				change_variable = {
					which = insyaa_charkuchin_kobold_citadel_supplies_supply_facility_count
					value = 1
				}
				change_variable = {
					which = insyaa_charkuchin_kobold_citadel_supplies_total_gain
					value = 10
				}
			}
		}
	}
	if = { # Otherwise, it's a province we need to spend monthly supplies on. This will include provinces that had a SF that we just deleted, hence "if" instead of "else"
		limit = {
			NOT = {
				province_group = citadel_group_main # Should be unnecessary, but just in case
				has_province_modifier = charkuchin_kobold_supply_facility
				AND = {
					owner = { exists = yes }
					is_city = no
				}
			}
		}
		if = { # Different regions have different costs so we track them differently
			limit = { superregion = charkuchar_superregion }
			owner = {
				insyaa_charkuchin_kobold_citadel_supplies_update_counter = {
					region_group = charkuchar_range
					location_type = city
					value = 1
					cost = 1.5
				}
			}
		}
		else_if = {
			limit = {
				superregion = insyaan_lowlands_superregion
				NOT = { region = tagaala_mangroves_region }
			}
			owner = {
				insyaa_charkuchin_kobold_citadel_supplies_update_counter = {
					region_group = lowlands
					location_type = city
					value = 1
					cost = 6
				}
			}
		}
		else_if = {
			limit = {
				superregion = hinuilands_superregion
			}
			owner = {
				insyaa_charkuchin_kobold_citadel_supplies_update_counter = {
					region_group = titanoflora
					location_type = city
					value = 1
					cost = 30
				}
			}
		}
	}
}

insyaa_charkuchin_kobold_citadel_supplies_update_counter = {
	change_variable = {
		which = insyaa_charkuchin_kobold_citadel_supplies_$region_group$_$location_type$_count
		value = $value$
	}

	change_variable = {
		which = insyaa_charkuchin_kobold_citadel_supplies_$region_group$_$location_type$_cost
		value = $cost$
	}
	# Update the total cost in here, too
	change_variable = {
		which = insyaa_charkuchin_kobold_citadel_supplies_total_cost
		value = $cost$
	}
	#The total cost is tripled during the migration
	if = {
		limit = {
			has_disaster = hinuibird_migration_disaster
		}
		# Update the total cost in here, too
		change_variable = {
			which = insyaa_charkuchin_kobold_citadel_supplies_total_cost
			value = $cost$
		}
		# Update the total cost in here, too
		change_variable = {
			which = insyaa_charkuchin_kobold_citadel_supplies_total_cost
			value = $cost$
		}
	}
}

insyaa_charkuchin_kobold_citadel_supplies_monthly_loss = {
	set_variable = {
		which = insyaa_charkuchin_kobold_citadel_supplies_total_cost_temp
		which = insyaa_charkuchin_kobold_citadel_supplies_total_gain
	}
	subtract_variable = {
		which = insyaa_charkuchin_kobold_citadel_supplies_total_cost_temp
		which = insyaa_charkuchin_kobold_citadel_supplies_total_cost
	}
	if = {
		limit = { # If it's positive
			check_variable = {
				which = insyaa_charkuchin_kobold_citadel_supplies_total_cost_temp
				value = 0
			}
		}
		gov_power_var_effect = {
			mechanic = charkuchin_kobold_citadel_supplies_mechanic
			power = citadel_supplies
			var = insyaa_charkuchin_kobold_citadel_supplies_total_cost_temp
			sign = +
		}
	}
	else = { # gov_power_var_effect only works with positive numbers, so it our value if negative, flip it and use sign = -
		multiply_variable = {
			which = insyaa_charkuchin_kobold_citadel_supplies_total_cost_temp
			value = -1
		}
		gov_power_var_effect = {
			mechanic = charkuchin_kobold_citadel_supplies_mechanic
			power = citadel_supplies
			var = insyaa_charkuchin_kobold_citadel_supplies_total_cost_temp
			sign = -
		}
	}
}

insyaa_charkuchin_kobolds_released_setup = {
	clr_country_flag = charkuchin_kobold_start #I'm pretty sure flags stick around if you are annexed then released

	# Since your government type depends on the type of the country that released you as a vassal or forced you to be
	# released via peace, and we ony have 7 starting Insyaa kobold tags, I'll hardcode their starting gov types
	# If you're not in that list (maybe more kobolds in the future) you get a monarchy if you were native
	# or tribal
	trigger_switch = {
		on_trigger = tag
		P30 = { change_government = monarchy }
		P31 = { change_government = theocracy }
		P32 = { change_government = monarchy }
		P33 = { change_government = monarchy }
		P34 = { change_government = monarchy }
		P35 = { change_government = republic }
		P36 = { change_government = republic }
	}
	trigger_switch = {
		on_trigger = government
		monarchy = { add_government_reform = charkuchin_monarchy_reform }
		republic = { add_government_reform = charkuchin_council_reform }
		theocracy = { add_government_reform = charkuchin_citadel_clerical_administration }
		tribal = {
			change_government = monarchy
			add_government_reform = charkuchin_monarchy_reform
		}
		native = {
			change_government = monarchy
			add_government_reform = charkuchin_monarchy_reform
		}
	}
	country_event = { #Initial variable setup and some modifiers
		id = charkuchin_kobold.0
	}
	#Update the supplies counters for your provinces, since the starting event assumes
	#you only have your capital with a citadal
	every_owned_province = {
		limit = {
			is_capital = no
		}
		insyaa_charkuchin_kobold_citadel_supplies_update_count_add = yes
	}
	#Update your gain from admin tech
	export_to_variable = {
		which = kobold_admin_tech
		value = adm_tech
	}
	#Since the starting event assumes tech 2
	subtract_variable = { kobold_admin_tech = 2 }
	while = {
		limit = {
			check_variable = {
				which = kobold_admin_tech
				value = 1
			}
		}
		subtract_variable = { kobold_admin_tech = 1 }
		insyaa_charkuchin_kobold_citadel_supplies_tech = yes
	}
}

kobold_citadel_food_button = {
	if = { #Disable Ordinance
		limit = { has_province_flag = current_ordinance_food }
		if = {
			limit = {
				has_province_flag = upgraded_ordinance_food
			}
			kobold_citadel_clear_ordinance_food = { name = $name$ upgraded = yes }
		}
		else = {
			kobold_citadel_clear_ordinance_food = { name = $name$ }
		}
	}
	else_if = { #After Upgrade
		limit = {
			has_province_flag = upgraded_ordinance_food
		}
		custom_tooltip = ordinance_durin_active_food_tt
		kobold_citadel_any_group_effect_food = { name = $name$ upgraded = yes }
	}
	else = { #Starting version
		custom_tooltip = ordinance_durin_active_food_tt
		kobold_citadel_any_group_effect_food = { name = $name$ }
		#Tooltip
		custom_tooltip = ordinance_durin_upgrade_tt
		tooltip = {
			kobold_citadel_any_group_upgrading_effect = { name = $name$ ordinance = food }
		}
		custom_tooltip = ordinance_after_upgrade_tt
		kobold_citadel_any_group_effect_food_tooltip = { name = $name$ }
	}
}

export_build_cost_and_time = {
	export_to_variable = {
		which = globalCost
		value = modifier:build_cost
		who = owner
	}
	export_to_variable = {
		which = globalTime
		value = modifier:build_time
		who = owner
	}
	export_to_variable = {
		which = localCost
		value = modifier:local_build_cost
	}
	export_to_variable = {
		which = localTime
		value = modifier:local_build_time
	}
	change_variable = {
		which = localCost
		which = globalCost
	}
	change_variable = {
		which = localTime
		which = globalTime
	}
}
kobold_create_tooltip_for_building = {
	multiply_variable = {
		which = localTime
		value = $base_time$
	}
	multiply_variable = {
		which = localCost
		value = $base_cost$
	}
}
cap_build_cost_and_time = {
	#
	if = {
		limit = { NOT = { check_variable = { localTime = $time_cap$ } } }
		set_variable = { localTime = $time_cap$ }
	}
	
	if = {
		limit = { NOT = { check_variable = { localCost = $cost_cap$ } } }
		set_variable = { localCost = $cost_cap$ }
	}
	#
}

### Construction of fake building that self destruct after construction finish ###
kobold_citadel_add_construction = {
	if = {
		limit = {
			NOT = { check_variable = { local_build_cost = -0.9 } } #If you have more than 90% reduction of cost just eat 10% of cost
		}
		set_variable = {
			cappedcost = $cost$
		}
		multiply_variable = { which = cappedcost value = 0.1 }
		while = {
			limit = {
				check_variable = {
					which = cappedcost
					value = 1
				}
			}
			subtract_variable = {
				which = cappedcost
				value = 1
			}
			add_treasury = -1
		}
	}
	#
	add_building_construction = { building = kobold_citadel_expansion speed = $speed$ cost = $cost$ }
}

kobold_citadel_ai_upgrade_citadel = {
	if = {
		limit = {
			has_province_flag = current_ordinance_food
		}
		if = { #Upgrade Food
			limit = {
				NOT = { has_province_flag = upgraded_ordinance_food }
			}
			kobold_citadel_clear_ordinance_food = { name = $name$ }
			set_province_flag = upgrading_ordinance_food
		}
		else = {
			if = { #If they are able to finish roc system add version without supply to this effect
				limit = { has_province_flag = kobold_citadel_level_2 }
				remove_province_modifier = charkuchin_kobold_ordinance_food_main_1_upgraded
				add_permanent_province_modifier = {
					name = charkuchin_kobold_ordinance_food_main_2_upgraded
					duration = -1
				}
				change_variable = {
					which = insyaa_charkuchin_kobold_citadel_supplies_ordinance_gain
					value = 5
				}
				change_variable = {
					which = insyaa_charkuchin_kobold_citadel_supplies_total_gain
					value = 5
				}
			}
		}
	}
	#Upgrade Education
	if = {
		limit = {
			has_province_flag = current_ordinance_education
			NOT = { has_province_flag = upgraded_ordinance_education }
		}
		kobold_citadel_clear_ordinance_education = { name = $name$ }
		set_province_flag = upgrading_ordinance_education
	}
	
	if = {
		limit = {
			has_province_flag = current_ordinance_foundations
		}
		if = { #Upgrade Foundations
			limit = { NOT = { has_province_flag = upgraded_ordinance_foundations } }
			kobold_citadel_clear_ordinance_foundations = { name = $name$ }
			set_province_flag = upgrading_ordinance_foundations
		}
		else = {
			set_province_flag = ai_kobold_citadel_update_mountain
		}
	}
	#Upgrade Working
	if = {
		limit = {
			has_province_flag = current_ordinance_working
			NOT = { has_province_flag = upgraded_ordinance_working }
		}
		kobold_citadel_clear_ordinance_working = { name = $name$ }
		set_province_flag = upgrading_ordinance_working
	}
	#Upgrade Services
	if = {
		limit = {
			has_province_flag = current_ordinance_services
			NOT = { has_province_flag = upgraded_ordinance_services }
		}
		kobold_citadel_clear_ordinance_services = { name = $name$ }
		set_province_flag = upgrading_ordinance_services
	}
	remove_province_modifier = ai_have_enacted_unupgraded_ordinance
	upgrade_kobold_citadel = { name = $name$ }
	if = {
		limit = { has_province_flag = ai_kobold_citadel_update_mountain }
		clr_province_flag = ai_kobold_citadel_update_mountain
		kobold_citadel_foundations_mountain = { name = $name$ }
	}
}
