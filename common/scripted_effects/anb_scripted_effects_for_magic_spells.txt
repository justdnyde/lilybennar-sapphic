change_patron_approval = {
	[[add]
		custom_tooltip = patron_pleased_tt
		hidden_effect = {
			change_variable = { PatronApproval = $add$ }
			if = { limit = { check_variable = { PatronApproval = 9 } }
				set_variable = { PatronApproval = 8 }
			}
		}
	]
	[[remove]
		custom_tooltip = patron_displeased_tt
		hidden_effect = {
			subtract_variable = { PatronApproval = $remove$ }
			if = { limit = { NOT = { check_variable = { PatronApproval = 0 } } }
				set_variable = { PatronApproval = 0 }
			}
		}
	]
}

### Spells ###

### THE HIERARCHY IS AS FOLLOWS:
### A SCHOOL DEFINES UNLOCK PATTERNS AND OTHER GROUPINGS FOR SPELLS (AKA YOU WILL UNLOCK CONJURATION 2 AFTER CONJURATION 1)
### A LEVEL IS WHAT DEFINES THE COST OF THE SPELL IN MANA: WE RESPECT FOUR LEVELS ONLY (0, 1, 2, 3) THOUGH YOU CAN MAKE MORE
### AN ID DISTINGUISHES BETWEEN DIFFERENT SPELLS OF THE SAME SCHOOL AND LEVEL
### EXAMPLE (abjuration_2_mass_ward_effect) and (abjuration_2_mage_armor_effect). BOTH ABJURATION, BOTH LEVEL 2, DIFFERENT ID'S

make_spell_free = {
	[[ruler]
		custom_tooltip = $school$_$level$_$id$_spell_free
		custom_tooltip = while_ruler_alive_tt
		set_ruler_flag = $school$_$level$_$id$_spell_free
	]
	[[heir]
		custom_tooltip = $school$_$level$_$id$_spell_free
		custom_tooltip = while_heir_alive_tt
		set_heir_flag = $school$_$level$_$id$_spell_free
	]
	[[next_cast]
		custom_tooltip = $school$_$level$_$id$_spell_free
		custom_tooltip = for_next_cast_only_tt
		set_country_flag = $school$_$level$_$id$_spell_free_next_cast
	]
	[[duration]
		add_country_modifier = { name = $school$_$level$_$id$_spell_free duration = $duration$ }
	]
}

generic_cast_spell = { #Duplicate relevant changes in 'magical_tradition_cast_spell'

	# Handles paying mana, given a limited number of ways to alter mana cost
	if = {
		limit = { has_mana_discount = { school = $school$ level = $level$ id = $id$ discount = free } }

		custom_tooltip = spell_is_free_tt
		remove_discount_next_cast_flag = { school = $school$ level = $level$ id = $id$ discount = free }
	}
	else_if = { #Half-cost spell
		limit = { has_mana_discount = { school = $school$ level = $level$ id = $id$ discount = half } }

		custom_tooltip = spell_is_half_tt
		remove_discount_next_cast_flag = { school = $school$ level = $level$ id = $id$ discount = half }

		pay_half_mana_for_level_$level$_spell = { school = $school$ }
	}
	else_if = { #25% off spell
		limit = { has_mana_discount = { school = $school$ level = $level$ id = $id$ discount = three_quarter } }

		custom_tooltip = spell_is_three_quarter_tt
		remove_discount_next_cast_flag = { school = $school$ level = $level$ id = $id$ discount = three_quarter }

		pay_three_quarter_mana_for_level_$level$_spell = { school = $school$ }
	}
	else = { pay_mana_for_level_$level$_spell = { school = $school$ } }

	[[war]
		custom_tooltip = only_one_war_magic_tt
		hidden_effect = { clear_active_war_magic = yes }
		on_war_magic_cast = yes
	]

	hidden_effect = {
		count_spells_effect = { school = $school$ level = $level$ id = $id$ }
	}
	
	on_$school$_spell_cast = yes

	on_level_$level$_spell_cast = yes

	$school$_$level$_$id$_effect = yes
}
remove_discount_next_cast_flag = {
	trigger_switch = { #Next cast discounted spell handled inside
		on_trigger = has_country_flag

		$school$_$level$_$id$_spell_$discount$_next_cast = { custom_tooltip = for_next_cast_only_tt clr_country_flag = $school$_$level$_$id$_spell_$discount$_next_cast }
		$school$_spell_$discount$_next_cast = { custom_tooltip = for_next_cast_only_tt clr_country_flag = $school$_spell_$discount$_next_cast }
		all_spell_$discount$_next_cast = { custom_tooltip = for_next_cast_only_tt clr_country_flag = all_spell_$discount$_next_cast }
	}
}

count_spells_effect = { #school/all, internal use
	if = {
		limit = { has_country_flag = count_$school$_spells }
		change_variable = { which = $school$_spell_count value = 1 }
	}
	if = {
		limit = { has_country_flag = count_all_spells }
		change_variable = { which = all_spell_count value = 1 }
	}
	if = {
		limit = { has_country_flag = count_$school$_$level$_$id$_spell }
		change_variable = { which = $school$_$level$_$id$_spell_count value = 1 }
	}
}

#### Instructions for new spells/spell lists

# Your Magical Traditions 'spell list' must have a name, consisting of:
#	- a 'group': the name of the custom magic group
#	- a 'subgroup': the specific name of your list
# ( the base list in each custom magic group must have subgroup = default )
# This will be used across all scripted effects and triggers

# All new spell lists require the following:
# - A new entry in the respective 'clr_<group>_spell_lists' scripted effect
# - A new scripted trigger 'has_<group>_spell_list_<subgroup>' in the respective 'Pre-formed magical tradition triggers' scripted effect
# - Localisation for spell list name, tooltips, etc, in localisation\anb_magic_spells_custom_l_english.yml
# - Custom localization: entries in customizable_localization\anb_magic_menus_localization.txt
#	- 'getTraditionNameGroup<group>' and 'getTraditionName<Group><Subgroup>' sections

# New spells each require the following:
# - Scripted effects in the 'Magical Traditions scripted effects' section of the file
#	- named with the format: <school>_<level>_<group>_<subgroup>_<spellnumber>_effect
# - Scripted triggers in the 'Magical traditions spell triggers' section
#	- Requires both:
#		- custom_spell_<group>_<subgroup>_<spellnumber>_trigger
#		- $school$_$level$_$id$_trigger = yes -- this enables using a spell in multiple spell lists
# - Localisation for spell names, tooltips, etc, in localisation\anb_magic_spells_custom_l_english.yml
# - Custom localization: entries in customizable_localization\anb_magic_menus_localization.txt
#	- Find the '# Spell list custom loc' section
# - Modifiers:
#	- Put these in the 'MAGICAL TRADITIONS' section in common\event_modifiers\anb_magic_modifiers.txt

			]
		}
	}
}
