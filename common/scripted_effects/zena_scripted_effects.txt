zena_relic_purchase_effect = { #Buuying relics
	if = {
		limit = {
			check_variable = {
				which = zena_blueprints_of_tomorrow_value
				value = 15
			}
		}
		set_variable = {
			which = zena_relic_cost
			value = 0
		}
	}
	else = {
		export_to_variable = {
			which = zena_relic_cost
			value = monthly_income
			who = ROOT 
		}
		set_variable = {
			which = zena_relic_multiplier
			value = 18
		}
		if = { # Subtract for opinion
			limit = {
				event_target:$zena_relic_event_target$ = {
					has_opinion = {
						who = ROOT
						value = 100
					}
				}
			}
			subtract_variable = {
				which = zena_relic_multiplier
				value = 3
			}
		}
		else_if = {
			limit = {
				event_target:$zena_relic_event_target$ = {
					has_opinion = {
						who = ROOT
						value = 25
					}
				}
			}
			subtract_variable = {
				which = zena_relic_multiplier
				value = 1
			}
		}
		if = { #-2 months if subject
			limit = {
				event_target:$zena_relic_event_target$ = {
					is_subject_of = ROOT
				}
			}
			subtract_variable = {
				which = zena_relic_multiplier
				value = 2
			}
		}
		if = { #-1 month if allied
			limit = {
				event_target:$zena_relic_event_target$ = {
					alliance_with = ROOT
				}
			}
			subtract_variable = {
				which = zena_relic_multiplier
				value = 1
			}
		}
		if = { 
			limit = {
				is_ruler_enchantment_level_3 = yes
			}
			subtract_variable = {
				which = zena_relic_multiplier
				value = 4
			}
		}
		else_if = {
			limit = {
				is_ruler_enchantment_level_1_minimum = yes
			}
			subtract_variable = {
				which = zena_relic_multiplier
				value = 2
			}
		}
		if = { 
			limit = {
				is_estate_enchantment_level_3 = yes
			}
			subtract_variable = {
				which = zena_relic_multiplier
				value = 4
			}
		}
		else_if = {
			limit = {
				is_estate_enchantment_level_1_minimum = yes
			}
			subtract_variable = {
				which = zena_relic_multiplier
				value = 2
			}
		}
		if = { #-3 months for completing 'the worlds collection'
			limit = {
				mission_completed = L06_worlds_collection
			}
			subtract_variable = {
				which = zena_relic_multiplier
				value = 3
			}
		}
		multiply_variable = {
			which = zena_relic_cost
			which = zena_relic_multiplier
		}
	}
}
