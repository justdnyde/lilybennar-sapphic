	sprite = "GFX_gather_great_host_great"
	cost = 200
	trigger = {
		custom_trigger_tooltip = {
			tooltip = irdaeos_aspect_tournament_tt
			OR = {
			has_country_flag = irdaeos_tournament_scarrification
			has_country_flag = irdaeos_tournament_headhunting
			has_country_flag = irdaeos_tournament_drawn
			}
		}
		OR = {
			custom_trigger_tooltip = {
				tooltip = irdaeos_celebration_in_action_tt
				NOT = { has_country_modifier = irdaeos_celebration_cooldown }
			}
			irdaeos_worship_religious_action_cooldown_bypass = { type = filustkhuga }
		}
	}
	
	effect = {
		country_event = { id = irdaeos_worship.101 }
		set_country_flag = irdaeos_celebration
		if = {
			limit = {
				has_country_flag = I69_filust_2
				has_estate_privilege = estate_mages_I69_codification_of_our_consensus
			}
			add_yearly_manpower = 1
		}
		hidden_effect = {
			I69_religious_action_murdering = { school = filustkhuga_school } #Star Rider things
			I69_festivals_irdaeos = yes
			if = {
				limit = { 
					has_country_flag = irdaeos_worship_ignore_scholar_filustkhuga
					NOT = { 
						has_country_modifier = filustkhuga_scholar_modifier
						religious_school = {
							group = centaur_religion
							school = filustkhuga_school
						} 		
					} 
				}
				clr_country_flag = irdaeos_worship_ignore_scholar_filustkhuga
			}
			if = {
				limit = { has_country_modifier = irdaeos_celebration_cooldown }
				if = {
					limit = { has_country_flag = irdaeos_worship_ignore_filustkhuga_cooldown_once }
					clr_country_flag = irdaeos_worship_ignore_filustkhuga_cooldown_once
				}
				else_if = {
					limit = { has_country_flag = irdaeos_worship_ignore_all_cooldown_once }
					clr_country_flag = irdaeos_worship_ignore_all_cooldown_once
				}
			}
			add_country_modifier = {
				name = irdaeos_celebration_cooldown
				duration = 3650
				hidden = yes
			}
		}
	}
	
	ai_will_do = {
		factor = 100
	}
}
