speaking_memory_aspect_listen_to_dragon_turtle = {
	sprite = "GFX_sky_domain_amilak_supremacy"
	cost = 100
	potential = {
	}
	trigger = {
		hidden_trigger = { NOT = { has_country_flag = speaking_memory_event_open } }
		any_owned_province = {
			province_has_dragon_turtle = yes
			NOT = { has_province_modifier = speaking_memory_listening_to_stories }
			NOT = { has_province_modifier = speaking_memory_asking_expertise }
		}
	}

	effect = {
		hidden_effect = { set_country_flag = speaking_memory_event_open }
		country_event = {
			id = flavour_speaking_memory.100
		}
	}

	ai_will_do = {
		factor = 1
		modifier = {
			factor = 20
			NOT = {
				check_variable = {
					which = speaking_memory_ancestory_percentage
					value = 50
				}
			}
		}
		modifier = {
			factor = 20
			NOT = {
				check_variable = {
					which = speaking_memory_ancestory_percentage
					value = 20
				}
			}
		}
	}
}

speaking_memory_aspect_ancestral_emulation = {
	sprite = "GFX_sky_domain_great_prayer"
	cost = 100
	potential = {
	}
	trigger = {
		hidden_trigger = { NOT = { has_country_flag = speaking_memory_event_open } }
		custom_trigger_tooltip = {
			tooltip = ancestral_emulation_cooldown_tt
			NOT = { has_country_modifier = speaking_memory_ancestral_emulation_cooldown }
		}
		custom_trigger_tooltip = {
			tooltip = ancestral_emulation_cooldown_2_tt
			NOT = {
				AND = {
					adm = 6
					dip = 6
					mil = 6
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = speaking_memory_need_15_memory_tt
			check_variable = {
				which = speaking_memory_ancestory_percentage
				value = 15
			}
		}
	}

	effect = {
		hidden_effect = {
			set_country_flag = speaking_memory_event_open
			random_list = {
				1 = {
					trigger = { NOT = { adm = 6 } }
					set_country_flag = sm_emulation_adm
				}
				1 = {
					trigger = { NOT = { dip = 6 } }
					set_country_flag = sm_emulation_dip
				}
				1 = {
					trigger = { NOT = { mil = 6 } }
					set_country_flag = sm_emulation_mil
				}
			}
			add_country_modifier = {
				name = speaking_memory_ancestral_emulation_cooldown
				duration = 3650
				hidden = yes
			}
		}
		country_event = {
			id = flavour_speaking_memory.104
		}
	}

	ai_will_do = {
		factor = 15
		modifier = {
			factor = 0.5
			NOT = {
				check_variable = {
					which = speaking_memory_ancestory_percentage
					value = 50
				}
			}
		}
		modifier = {
			factor = 0.5
			NOT = {
				check_variable = {
					which = speaking_memory_ancestory_percentage
					value = 20
				}
			}
		}
	}
}
