charkuchin_recruits = {
	sprite = "GFX_dissolve_monasteries_icon"
	cost = 100
	trigger = {
		any_known_country = {
			culture_is_kobold = yes
			if = {
				limit = { ROOT = { ai = no } }
				has_opinion = {
					who = ROOT
					value = 50
				}
			}
			else = {
				has_opinion = {
					who = ROOT
					value = 10
				}
			}
			NOT = {
				has_opinion_modifier = {
					who = ROOT
					modifier = dunborgula_positive_opinion
				}
			}
		}
	}
	effect = {
		custom_tooltip = charkuchin_recruits_tt
		hidden_effect = {
			add_manpower = 5
			add_treasury = 50
			random_known_country = {
				limit = {
					culture_is_kobold = yes
					if = {
						limit = { ROOT = { ai = no } }
						has_opinion = {
							who = ROOT
							value = 50
						}
					}
					else = {
						has_opinion = {
							who = ROOT
							value = 10
						}
					}
					NOT = {
						has_opinion_modifier = {
							who = ROOT
							modifier = dunborgula_positive_opinion
						}
					}
				}
				add_opinion = {
					who = ROOT
					modifier = dunborgula_positive_opinion
				}
				if = {
					limit = {
						ROOT = {
							OR = { 
								has_danguranga_water_earth_school = yes
								has_aycha_alpa_school = yes
							}
						}
					}
					if = {
						limit = { adm_tech_is_inferior = { tag = ROOT } }
						ROOT = { add_adm_power = 25 }
					}
					if = {
						limit = { dip_tech_is_inferior = { tag = ROOT } }
						ROOT = { add_dip_power = 25 }
					}
					if = {
						limit = { mil_tech_is_inferior = { tag = ROOT } }
						ROOT = { add_mil_power = 25 }
					}
				}
			}
		}
		if = {
			limit = {
				OR = {
					has_danguranga_water_earth_school = yes
					has_aycha_alpa_school = yes
				}
			}
			custom_tooltip = charkuchin_recruits_bonus_tt_YES
		}
		else = {
			custom_tooltip = charkuchin_recruits_bonus_tt_NO
		}
	}
	ai_will_do = {
		factor = 1
		modifier = {
			factor = 2
			church_power = 150
		}
		modifier = {
			factor = 2
			is_at_war = yes
		}
		modifier = {
			factor = 3
			num_of_loans = 2
		}
		modifier = {
			factor = 2
			NOT = { treasury = 50 }
		}
		modifier = {
			factor = 2
			NOT = { manpower = 5 }
		}
	}
}

					subtract_variable = {
						which = ForLoopCounter
						value = 1
					}
				}
			}
			add_country_modifier = {
				name = yenbenmirra_the_fightback_cd
				duration = 1825

			}
		}
		if = {
			limit = {

			}
