
tag_can_use_mechanim_ward_conduit = {
	# Mechanim formables
	OR = {
		tag = M22 # Kiretua
		tag = M23 # Rangitema
		tag = M24 # Anahira
		tag = M25 # Pa-Aroha
		tag = M26 # Rakauma
	}
}

province_can_get_mechanim_ward_conduit = {
	if = {
		limit = { owner = { tag = M22 } } # Kiretua
		owner = { tag = M22 }
		OR = {
			area = titanoflora_lakelands_5_area
			area = titanoflora_lakelands_6_area
		}
	}
	else_if = {
		limit = { owner = { tag = M23 } } # Rangitema
		owner = { tag = M23 }
		OR = {
			area = titanoflora_lakelands_10_area
			area = titanoflora_lakelands_11_area
			area = titanoflora_lakelands_12_area
			area = titanoflora_lakelands_13_area
		}
	}
	else_if = {
		limit = { owner = { tag = M24 } } # Anahira
		owner = { tag = M24 }
		OR = {
			area = titanoflora_riverlands_11_area
			area = titanoflora_riverlands_13_area
			province_id = 3496
			province_id = 3331
			province_id = 3495
		}
	}
	else_if = {
		limit = { owner = { tag = M25 } } # Pa-Aroha
		owner = { tag = M25 } # Pa-Aroha
		OR = {
			area = titanoflora_riverlands_16_area
			area = titanoflora_riverlands_17_area
			province_id = 3183
			province_id = 3181
			province_id = 3322
			province_id = 3323
		}
	}
	else_if = {
		limit = { owner = { tag = M26 } } # Rakauma
		owner = { tag = M26 }
		OR = {
			area = turuputai_area
			area = manuata_area
			province_id = 3159
			province_id = 3160
		}
	}
	else = {
		# unreachable because this is locked behind tag_can_use_mechanim_ward_conduit
		always = no
	}
}

dragon_turtle_matched_relationship_flag_trigger = { #do you have the right flag for your relationship value?
	if = {
		limit = { check_variable = { dragon_turtle_friendship_$dragon_turtle$ = 361 } }
		has_country_flag = beloved_$dragon_turtle$
	}
	else_if = {
		limit = { check_variable = { dragon_turtle_friendship_$dragon_turtle$ = 301 } }
		has_country_flag = trusted_$dragon_turtle$
	}
	else_if = {
		limit = { check_variable = { dragon_turtle_friendship_$dragon_turtle$ = 241 } }
		has_country_flag = friendly_$dragon_turtle$
	}
	else_if = {
		limit = { check_variable = { dragon_turtle_friendship_$dragon_turtle$ = 181 } }
		has_country_flag = neutral_$dragon_turtle$
	}
	else_if = {
		limit = { check_variable = { dragon_turtle_friendship_$dragon_turtle$ = 121 } }
		has_country_flag = disapproving_$dragon_turtle$
	}
	else_if = {
		limit = { check_variable = { dragon_turtle_friendship_$dragon_turtle$ = 61 } }
		has_country_flag = hated_$dragon_turtle$
	}
	else = { has_country_flag = reviled_$dragon_turtle$ }
}
