has_voltoj_academy = { # Checks on Province Level whatever there is an academy of any size (hold or not)
	custom_trigger_tooltip = {
		tooltip = overclan_has_voltoj_academy_tt
		province_group = voltoj_city_province
		OR = {
			has_province_modifier = F78_small_voltoj_academy
			has_province_modifier = F78_small_voltoj_academy_hold
			has_province_modifier = F78_medium_voltoj_academy
			has_province_modifier = F78_medium_voltoj_academy_hold
			has_province_modifier = F78_large_voltoj_academy
			has_province_modifier = F78_large_voltoj_academy_hold
		}
	}
}

voltoj_can_afford_upgrade_to_medium = {
	has_voltoj_academy = yes
	has_major_voltoj_academy = no
	owner = { has_enacted_parliament_action = { parliament_action = overclan_expand_voltoj_recruitment_act } }
	if = {
		limit = { province_group = hold_province }
		development = 50
	}
	else = {
		development = 30
	}
}
voltoj_can_afford_upgrade_to_large = {
	has_medium_voltoj_academy = yes
	if = {
		limit = { province_group = hold_province }
		development = 60
	}
	else = {
		development = 40
	}
}

overclan_unique_upgrade_available = { #province scope
	province_group = voltoj_city_province
	OR = {
		OR = {
			#no specific unlock
			province_id = 521 #re-uyel
			province_id = 2942 #shaz
			province_id = 4124 #segdihr
		}
		AND = {
			province_id = 526 #OT
			owner = { has_country_flag = F78_inscription_buildings_unlocked }
		}
		AND = {
			voltoj_province_in_dwarven_group = yes
			owner = { has_issue_unlocked = { issue = overclan_sponsor_dwarven_settlement } }
		}
		AND = {
			voltoj_province_in_artificery_group = yes
			owner = { has_country_flag = F78_artificery_buildings_unlocked }
		}
		AND = {
			voltoj_province_in_light_group = yes
			owner = { has_country_flag = F78_light_buildings_unlocked }
		}
		AND = {
			voltoj_province_in_order_group = yes
			owner = { has_issue_unlocked = { issue = overclan_send_out_judges } }
		}
		AND = {
			province_id = 631 #eduz vac
			owner = { has_country_flag = F78_ravelian_buildings_unlocked }
		}
	}
	has_unique_voltoj_building = no
}
