aPUI_check_variable = {
	[[value] check_variable = { which = aPUI_$name$_var value = $value$ } ]
	[[which] check_variable = { which = aPUI_$name$_var which = $which$ } ]
}
aPUI_is_variable_equal = {
	[[value] is_variable_equal = { which = aPUI_$name$_var value = $value$ } ]
	[[which] is_variable_equal = { which = aPUI_$name$_var which = $which$ } ]
}
#For comparing one aPUI var with another..
aPUI_check_aPUI_variable = {
	[[which] check_variable = { which = aPUI_$name$_var which = aPUI_$which$_var } ]
}
aPUI_is_aPUI_variable_equal = {
	[[which] is_variable_equal = { which = aPUI_$name$_var which = aPUI_$which$_var } ]
}
aPUI_check_capacity_variable = {
	[[which] check_variable = { which = aPUI_$name$_capacity which = $which$ } ]
	[[value] check_variable = { which = aPUI_$name$_capacity value = $value$ } ]
}
