#Open UI Button
custom_button = {
	name = aPUI_gene_pillar_interaction_button
	tooltip = aPUI_gene_pillar_interaction_button_tt
	potential = {
		has_eligible_aPUI_gene_pillar_interaction_in_province = yes
	}
	trigger = {
		if = {
			limit = { aPUI_gene_pillar_interaction_explored_all_levels = yes }
			custom_trigger_tooltip = {
				tooltip = aPUI_gene_pillar_interaction_explored_all_levels_tt
				always = no
			}
		}
		if = {
			limit = {
				FROM = {
					OR = {
						is_mechanim_tribe = yes
						government = native
					}
				}
			}
			FROM = {
				is_not_mechanim_tribe = yes
				NOT = { government = native }
			}
		}
	}
	effect = {
		open_aPUI_in_province = { name = gene_pillar_interaction }
		if = {
			limit = { owner = { NOT = { has_country_flag = aPUI_gene_pillar_interaction_setup_done } } }
			aPUI_gene_pillar_setup_first_time = yes
			# initial setup for the country and all owned gene pillars the first time this menu is opened
		}
		else = {
			aPUI_gene_pillar_setup_province = yes
			# failsafe - should be unnecessary? set up this province if it wasn't setup
		}
	}
	frame = {
		number = 1
		trigger = {
			OR = {
				province_id = 3329 #Orange - Hinuifauna
				province_id = 3173 #Yellow - Holohana
				province_id = 3153 #White - Forest
				province_id = 6950 #Pink - Collection
			}
		}
	}
	frame = {
		number = 2
		trigger = {
			OR = {
				province_id = 3376 #Red - Kiamoa
				province_id = 6962 #Gray - Kraken
			}
		}
	}
	frame = {
		number = 3
		trigger = {
			OR = {
				province_id = 3415 #Blue - Mechanim
				province_id = 6386 #Black - Dragon
			}
		}
	}
	frame = {
		number = 4
		trigger = {
			province_id = 3551 #Green - Sandworm
		}
	}
}

#The following 3 buttons serve both to start an expedition,
#AND to indicate how many expeditions have already been completed.
custom_button = {
	name = aPUI_gene_pillar_interaction_sections_button_1
	potential = { always = yes }
	trigger = {
		# If you are exploring this level, change what you see on the button triggers and effects
		if = {
			limit = {
				owner = {
					NOT = { has_country_flag = gene_pillar_is_exploring_level_1_pillar }
				}
			}
			aPUI_gene_pillar_interaction_not_explored_level_trigger = { level = 1 }
			is_not_exploring_any_gene_pillar = yes
			gene_pillars_has_any_class_selected = yes
			gene_pillars_did_not_party_wipe = yes
			gene_pillars_did_not_abandon_recently = yes
			if = {
				limit = { NOT = { controlled_by = owner } }
				controlled_by = owner
			}
		}
	}
	tooltip = gene_pillar_exploration_of_level_1_tt
	effect = {
		if = {
			limit = {
				owner = {
					NOT = { has_country_flag = gene_pillar_is_exploring_level_1_pillar }
				}
			}
			gene_pillar_start_exploration_of_level = { level = 1 }
		}
	}
	frame = { number = 1 }
}
