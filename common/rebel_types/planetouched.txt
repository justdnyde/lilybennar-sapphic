
######################################################################
# Planetouched rebels. They are planetouched.
######################################################################
planetouched_rebels = {
	
	color = { 117 53 5 }
	
	area = nation 					# Stays within the nation
	government = any				# Don't care about gov type				
	defection = none				# Will not defect
	independence = none				# Does not strive for independence
	gfx_type = culture_province
	will_relocate = yes	

	resilient = yes					# Can retreat
	reinforcing = no				# Can reinforce
	general = yes					# Does use generals
	smart = yes						# Do assault forts, tactically withdraw from battle, etc.
	unit_transfer = no				# Units do not remain after enforcing demands.
	
	artillery = 0.0
	infantry = 1.0
	cavalry = 0.0
	
	morale = 1.15
	
	# Possible handle actions
	handle_action_negotiate = yes
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 2.0
			is_capital = yes
		}
	}
	
 	# Province Scope
 	siege_won_trigger = {
 	}
	siege_won_effect = {
		add_local_autonomy = 25
	}	
	
	# Country scope
	can_negotiate_trigger = {
		always = yes
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}

	# Localisation for their demands
	demands_description = "particularist_rebels_demand"
	
	# Country scope
	demands_enforced_effect = {
		hidden_effect = {
			add_country_modifier = {
				name = just_lost_to_rebels
				duration = 10
				hidden = yes
			}
		}
		every_owned_province = {
			add_local_autonomy = 25
		}
		random_list = {
			1 = {
				trigger = { has_estate_privilege = estate_planetouched_scattered_scions }
				remove_estate_privilege = estate_planetouched_scattered_scions
			}
			1 = {
				trigger = { has_estate_privilege = estate_planetouched_scattered_banni }
				remove_estate_privilege = estate_planetouched_scattered_banni
			}
			1 = {
				trigger = { has_estate_privilege = estate_planetouched_scattered_commerce }
				remove_estate_privilege = estate_planetouched_scattered_commerce
			}
			1 = {
				trigger = { has_estate_privilege = estate_planetouched_scattered_jelli_sinful }
				remove_estate_privilege = estate_planetouched_scattered_jelli_sinful
			}
			1 = {
				trigger = { has_estate_privilege = estate_planetouched_scattered_unstable }
				remove_estate_privilege = estate_planetouched_scattered_unstable
			}
			1 = {
				trigger = { has_estate_privilege = estate_planetouched_scattered_titles }
				remove_estate_privilege = estate_planetouched_scattered_titles
			}
		}
	}
}
