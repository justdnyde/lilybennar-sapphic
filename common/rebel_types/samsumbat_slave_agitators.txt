######################################################################
# Slave uprising. Created during Samsumbat MT
######################################################################
samsumbat_slave_agitators = {

	color = { 50 85 210 }

	area = nation 					# Stays within the nation
	government = republic			
	defection = none				# Does not defect
	independence = none				# Does not strive for independence
	unit_transfer = yes
	gfx_type = culture_province	
	will_relocate = yes	
	
	resilient = yes
	reinforcing = yes
	general = yes
	smart = yes
	
	artillery = 0
	infantry = 0.9
	cavalry = 0.1
	
	morale = 1.1
	
	# Possible handle actions
	handle_action_negotiate = yes
	handle_action_stability = yes
	handle_action_build_core = yes
	handle_action_send_missionary = yes		
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1000
	
		modifier = {
			factor = 0
			NOT = { has_province_modifier = F60_slave_agitators_capital } # These will only spawn in the capital of countries given this modifier
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 10
			is_capital = yes
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
 	}
	siege_won_effect = {
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = yes
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = samsumbat_slave_agitators_demand
	
	# Country Scope	
	demands_enforced_effect = {
		add_prestige = -50
		hidden_effect = {
			add_country_modifier = {
				name = just_lost_to_rebels
				duration = 10
				hidden = yes
			}
		}
		every_owned_province = { # In case the capital moved
			remove_province_modifier = F60_slave_agitators_capital
		}
		custom_tooltip = F60_agitator_lose_subjects_tt
		hidden_effect = { # The tooltip makes it look like every *Samsumbati* subject will become independent :(
			release_all_subjects = yes
		}
		random_country = { # Since Samsumbat can form Prakheta without losing the MT
			limit = {
				is_or_was_tag = { tag = F60 }
			}
			samsumbat_slave_agitators_effect = yes
		}
		if = {
			limit = {
				NOT = { government = republic }
			}
			change_government = republic
		}
		add_government_reform = peasants_republic
	}
}
