warmth_rites_edict = {
	potential = {
		OR = {
			mission_completed = H14_decay_salvation
			AND = {
				overlord = {
					mission_completed = H14_decay_salvation
				}
				is_free_or_tributary_trigger = no
			}
		}
	}

	allow = {
		hidden_trigger = {
			OR = {
				mission_completed = H14_decay_salvation
				AND = {
					overlord = { mission_completed = H14_decay_salvation }
					is_free_or_tributary_trigger = no
				}
			}
		}
		FROM = {
			continent = north_america
			area_for_scope_province = {
				has_any_winter_trigger = yes
			}
			area_for_scope_province = { #To avoid needing to deal with gaining/losing provinces in the state in on-actions
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		ROOT = {
			if = {
                limit = {
                    ai = no
                    FROM = { has_any_warmth_rites_trigger = no } # so that mana is only required when placing the edict and not to maintain it
                }
                has_mana = { amount = 20 }        
            }
		}
	}

	modifier = { #Offset the normal edict maintenance modifier, and make the thing not empty lol
		local_state_maintenance_modifier = -2
	}

	color = { 244 123 32 }

	ai_will_do = {
		factor = 100
		modifier = {
			factor = 100
			OR = {
				province_id = 1191 #Svemel
				province_id = 1139 #Arverynn
				province_id = 1171 #Vareynn
				area = armonadh_area
				area = anholtir_area
			}
		}
	}

	on_state_edict_enacted = {
		owner = {
			if = {
				limit = {
					ai = no
				}
				change_mana = { amount = -20 }
			}
		}
		if = {
            limit = {
				OR = {
					owner = { has_country_flag = H14_warmth_rites_two }
					owner = { 
						overlord = { has_country_flag = H14_warmth_rites_two }
					}
				}
            }
			every_province_in_state = {
				add_province_modifier = {
					name = H14_warmth_rites_level_two
					duration = -1
				}
			}
			random_province_in_state = { 
				add_province_modifier = {
					name = H14_warmth_rites_hub_level_two
					duration = -1
				}
			}
        }
        else = {
			every_province_in_state = {
				add_province_modifier = {
					name = H14_warmth_rites
					duration = -1
				}
			}
			random_province_in_state = { 
				add_province_modifier = {
					name = H14_warmth_rites_hub
					duration = -1
				}
			}
        }
		if = {
			limit = {
				OR = {
					owner = { mission_completed = H14_secure_fringes }
					owner = {
						overlord = { mission_completed = H14_secure_fringes }
					}
				}
			}
			every_province_in_state = {
				limit = {
					has_fort_building_trigger = yes
				}
				add_province_modifier = {
					name = H14_secured_fringes
					duration = -1
				}
			}
		}
		if = {
			limit = {
				OR = {
					owner = { mission_completed = H14_magic_unrivaled }
					owner = {
						overlord = { mission_completed = H14_magic_unrivaled }
					}
				}
			}
			every_province_in_state = {
				limit = {
					has_building = mage_tower
				}
				add_province_modifier = {
					name = H14_magic_unrivaled
					duration = -1
				}
			}
		}
	}
	on_state_edict_revoked = {
		every_province_in_state = {
			add_devastation = 10
			remove_province_modifier = H14_warmth_rites_level_two
			remove_province_modifier = H14_warmth_rites
			remove_province_modifier = H14_warmth_rites_hub_level_two
			remove_province_modifier = H14_warmth_rites_hub
			remove_province_modifier = H14_secured_fringes
			remove_province_modifier = H14_magic_unrivaled
		}
	}
}
