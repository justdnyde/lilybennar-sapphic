overclan_church_and_state_act = {
	category = 6
	allow = {
		has_issue_unlocked = {
			issue = overclan_church_and_state_act
		}
		has_not_enacted_parliament_action = {
			parliament_action = overclan_church_and_state_act
		}
	}
	effect = {
		country_event = { id = flavour_overclan.40 }
		set_parliament_action_flag = {
			parliament_action = overclan_church_and_state_act
		}
	}
	ai_will_do = { factor = 100 }
	chance = { factor = 1000 }
	chance_of_decision = 0
}
