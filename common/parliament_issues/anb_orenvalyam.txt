	category = 1

	allow = {
		has_reform = U39_kavalali_theo_democracy
		OR = {
			ruler_is_not_silverspite = yes
			has_country_flag = U39_enable_rite_of_removal_flag
		}
	}

	effect = {
		if = {
			limit = { ruler_is_silverspite = yes }
			country_event = {
				id = orenvalyam_events.212 #The Duskbringer Conspiracy
			}
			add_country_modifier = {
				name = U39_careful_deliberation
				duration = -1
			}
		}
		else = {
			kill_ruler = yes
			if = {
				limit = { has_country_modifier = U39_ruler_monasticist }
				add_estate_loyalty = {
					estate = estate_burghers
					loyalty = -30
				}
			}
			else_if = {
				limit = { has_country_modifier = U39_ruler_kavalalist }
				add_estate_loyalty = {
					estate = estate_church
					loyalty = -30
				}
			}
			else_if = {
				limit = { has_country_modifier = U39_ruler_kamrayakvalist }
				add_estate_loyalty = {
					estate = estate_nobles
					loyalty = -30
				}
			}
		}
	}

	chance = {
		factor = 1000
	}
	
	ai_will_do = {
		factor = 10
	}
}
