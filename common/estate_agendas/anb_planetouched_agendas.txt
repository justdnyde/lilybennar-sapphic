

#Planetouched - Agenda New Military Host
estate_planetouched_military_host = {
	can_select = {
		has_estate = estate_planetouched
		NOT = { has_country_flag = anb_has_special_agendas }
		OR = {
			has_estate_privilege = estate_planetouched_organization_empowered
			has_estate_privilege = estate_planetouched_organization_state
		}
		mil_tech = 6
		num_of_owned_provinces_with = {
			value = 2
			NOT = { has_manpower_building_trigger = yes }
			num_free_building_slots = 1
		}
	}
	selection_weight = {
		factor = 1
	}
	pre_effect = {
		random_owned_province = {
			limit = {
				NOT = { has_manpower_building_trigger = yes }
				num_free_building_slots = 1
			}
			save_event_target_as = agenda_province_1
		}
		random_owned_province = {
			limit = {
				NOT = { has_manpower_building_trigger = yes }
				num_free_building_slots = 1
				NOT = { province_id = event_target:agenda_province_1 }
			}
			save_event_target_as = agenda_province_2
		}
	}
	immediate_effect = {
	}
	on_invalid = {
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_planetouched
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_planetouched
			}
		}
		else_if = {
			limit = { mil_tech = 16 }
			event_target:agenda_province_1 = { has_building = training_fields }
			event_target:agenda_province_2 = { has_building = training_fields }
		}
		else = {
			event_target:agenda_province_1 = { has_manpower_building_trigger = yes }
			event_target:agenda_province_2 = { has_manpower_building_trigger = yes }
		}
	}
	task_completed_effect = {
		on_completed_agenda_effect = { estate = estate_planetouched }
		add_estate_loyalty = {
			estate = estate_planetouched
			loyalty = 10
		}
		if = {
			limit = { army_tradition = 70 }
			define_planetouched_general = {
				fire = 5
				shock = 5
				manuever = 5
				siege = 3
			}
		}
		else_if = {
			limit = { army_tradition = 50 }
			define_planetouched_general = {
				fire = 4
				shock = 5
				manuever = 5
				siege = 2
			}
		}
		else_if = {
			limit = { army_tradition = 30 }
			define_planetouched_general = {
				fire = 3
				shock = 5
				manuever = 4
				siege = 1
			}
		}
		else_if = {
			limit = { army_tradition = 10 }
			define_planetouched_general = {
				fire = 3
				shock = 4
				manuever = 3
				siege = 1
			}
		}
		else = {
			define_planetouched_general = {
				fire = 2
				shock = 3
				manuever = 3
				siege = 1
			}
		}
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -10
		}
	}
	invalid_trigger = {
		OR = {
			event_target:agenda_province_1 = {
				NOT = { owned_by = ROOT }
			}
			event_target:agenda_province_2 = {
				NOT = { owned_by = ROOT }
			}
		}
	}
	failing_effect = {
		on_failed_agenda_effect = { estate = estate_planetouched }
		add_estate_loyalty_modifier = {
			estate = estate_planetouched
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
		}
	}
}

#Planetouched - Agenda Raise Earth Works
estate_planetouched_earth_works = {
	can_select = {
		has_estate = estate_planetouched
		NOT = { has_country_flag = anb_has_special_agendas }
		OR = {
			has_estate_privilege = estate_planetouched_organization_empowered
			has_estate_privilege = estate_planetouched_organization_state
		}
		any_owned_province = {
			NOT = { has_fort_building_trigger = yes }
			num_free_building_slots = 1
			any_neighbor_province = {
				NOT = { has_fort_building_trigger = yes }
				any_neighbor_province = {
					NOT = { has_fort_building_trigger = yes }
				}
			}
		}
	}
	selection_weight = {
		factor = 1
	}
	pre_effect = {
		random_owned_province = {
			limit = {
				NOT = { has_fort_building_trigger = yes }
				num_free_building_slots = 1
				any_neighbor_province = {
					NOT = { has_fort_building_trigger = yes }
					any_neighbor_province = {
						NOT = { has_fort_building_trigger = yes }
					}
				}
			}
			save_event_target_as = agenda_province
		}
	}
	immediate_effect = {
	}
	on_invalid = {
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_planetouched
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_planetouched
			}
		}
		else = {
			event_target:agenda_province = { province_has_current_tech_fort_trigger = yes }
		}
	}
	task_completed_effect = {
		on_completed_agenda_effect = { estate = estate_planetouched }
		add_estate_loyalty = {
			estate = estate_planetouched
			loyalty = 10
		}
		event_target:agenda_province = {
			add_country_modifier = {
				name = estate_planetouched_earth_works_mod
				duration = 3600 #10 years
			}
			add_base_manpower = 1
		}
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -10
		}
	}
	invalid_trigger = {
		event_target:agenda_province = {
			NOT = { owned_by = ROOT }
		}
	}
	failing_effect = {
		on_failed_agenda_effect = { estate = estate_planetouched }
		add_estate_loyalty_modifier = {
			estate = estate_planetouched
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
		}
	}
}
